228void CRYP_Init(CRYP_InitTypeDef* CRYP_InitStruct)
229{
230 /* Check the parameters */
231 assert_param(IS_CRYP_ALGOMODE(CRYP_InitStruct->CRYP_AlgoMode));
232 assert_param(IS_CRYP_DATATYPE(CRYP_InitStruct->CRYP_DataType));
233 assert_param(IS_CRYP_ALGODIR(CRYP_InitStruct->CRYP_AlgoDir));
234
235 /* Select Algorithm mode*/
236 CRYP->CR &= ~CRYP_CR_ALGOMODE;
237 CRYP->CR |= CRYP_InitStruct->CRYP_AlgoMode;
238
239 /* Select dataType */
240 CRYP->CR &= ~CRYP_CR_DATATYPE;
241 CRYP->CR |= CRYP_InitStruct->CRYP_DataType;
242
243 /* select Key size (used only with AES algorithm) */
244 if ((CRYP_InitStruct->CRYP_AlgoMode != CRYP_AlgoMode_TDES_ECB) &&
245 (CRYP_InitStruct->CRYP_AlgoMode != CRYP_AlgoMode_TDES_CBC) &&
246 (CRYP_InitStruct->CRYP_AlgoMode != CRYP_AlgoMode_DES_ECB) &&
247 (CRYP_InitStruct->CRYP_AlgoMode != CRYP_AlgoMode_DES_CBC))
248 {
249 assert_param(IS_CRYP_KEYSIZE(CRYP_InitStruct->CRYP_KeySize));
250 CRYP->CR &= ~CRYP_CR_KEYSIZE;
251 CRYP->CR |= CRYP_InitStruct->CRYP_KeySize; /* Key size and value must be
252 configured once the key has
253 been prepared */
254 }
255
256 /* Select data Direction */
257 CRYP->CR &= ~CRYP_CR_ALGODIR;
258 CRYP->CR |= CRYP_InitStruct->CRYP_AlgoDir;
259}

490ErrorStatus CRYP_SaveContext(CRYP_Context* CRYP_ContextSave,
491 CRYP_KeyInitTypeDef* CRYP_KeyInitStruct)
492{
493 __IO uint32_t timeout = 0;
494 uint32_t ckeckmask = 0, bitstatus;
495 ErrorStatus status = ERROR;
496
497 /* Stop DMA transfers on the IN FIFO by clearing the DIEN bit in the CRYP_DMACR */
498 CRYP->DMACR &= ~(uint32_t)CRYP_DMACR_DIEN;
499
500 /* Wait until both the IN and OUT FIFOs are empty
501 (IFEM=1 and OFNE=0 in the CRYP_SR register) and the
502 BUSY bit is cleared. */
503
504 if ((CRYP->CR & (uint32_t)(CRYP_CR_ALGOMODE_TDES_ECB | CRYP_CR_ALGOMODE_TDES_CBC)) != (uint32_t)0 )/* TDES */
505 {
506 ckeckmask = CRYP_SR_IFEM | CRYP_SR_BUSY ;
507 }
508 else /* AES or DES */
509 {
510 ckeckmask = CRYP_SR_IFEM | CRYP_SR_BUSY | CRYP_SR_OFNE;
511 }
512
513 do
514 {
515 bitstatus = CRYP->SR & ckeckmask;
516 timeout++;
517 }
518 while ((timeout != MAX_TIMEOUT) && (bitstatus != CRYP_SR_IFEM));
519
520 if ((CRYP->SR & ckeckmask) != CRYP_SR_IFEM)
521 {
522 status = ERROR;
523 }
524 else
525 {
526 /* Stop DMA transfers on the OUT FIFO by
527 - writing the DOEN bit to 0 in the CRYP_DMACR register
528 - and clear the CRYPEN bit. */
529
530 CRYP->DMACR &= ~(uint32_t)CRYP_DMACR_DOEN;
531 CRYP->CR &= ~(uint32_t)CRYP_CR_CRYPEN;
532
533 /* Save the current configuration (bit 19, bit[17:16] and bits [9:2] in the CRYP_CR register) */
534 CRYP_ContextSave->CR_CurrentConfig = CRYP->CR & (CRYP_CR_GCM_CCMPH |
535 CRYP_CR_KEYSIZE |
536 CRYP_CR_DATATYPE |
537 CRYP_CR_ALGOMODE |
538 CRYP_CR_ALGODIR);
539
540 /* and, if not in ECB mode, the initialization vectors. */
541 CRYP_ContextSave->CRYP_IV0LR = CRYP->IV0LR;
542 CRYP_ContextSave->CRYP_IV0RR = CRYP->IV0RR;
543 CRYP_ContextSave->CRYP_IV1LR = CRYP->IV1LR;
544 CRYP_ContextSave->CRYP_IV1RR = CRYP->IV1RR;
545
546 /* save The key value */
547 CRYP_ContextSave->CRYP_K0LR = CRYP_KeyInitStruct->CRYP_Key0Left;
548 CRYP_ContextSave->CRYP_K0RR = CRYP_KeyInitStruct->CRYP_Key0Right;
549 CRYP_ContextSave->CRYP_K1LR = CRYP_KeyInitStruct->CRYP_Key1Left;
550 CRYP_ContextSave->CRYP_K1RR = CRYP_KeyInitStruct->CRYP_Key1Right;
551 CRYP_ContextSave->CRYP_K2LR = CRYP_KeyInitStruct->CRYP_Key2Left;
552 CRYP_ContextSave->CRYP_K2RR = CRYP_KeyInitStruct->CRYP_Key2Right;
553 CRYP_ContextSave->CRYP_K3LR = CRYP_KeyInitStruct->CRYP_Key3Left;
554 CRYP_ContextSave->CRYP_K3RR = CRYP_KeyInitStruct->CRYP_Key3Right;
555
556 /* Save the content of context swap registers */
557 CRYP_ContextSave->CRYP_CSGCMCCMR[0] = CRYP->CSGCMCCM0R;
558 CRYP_ContextSave->CRYP_CSGCMCCMR[1] = CRYP->CSGCMCCM1R;
559 CRYP_ContextSave->CRYP_CSGCMCCMR[2] = CRYP->CSGCMCCM2R;
560 CRYP_ContextSave->CRYP_CSGCMCCMR[3] = CRYP->CSGCMCCM3R;
561 CRYP_ContextSave->CRYP_CSGCMCCMR[4] = CRYP->CSGCMCCM4R;
562 CRYP_ContextSave->CRYP_CSGCMCCMR[5] = CRYP->CSGCMCCM5R;
563 CRYP_ContextSave->CRYP_CSGCMCCMR[6] = CRYP->CSGCMCCM6R;
564 CRYP_ContextSave->CRYP_CSGCMCCMR[7] = CRYP->CSGCMCCM7R;
565
566 CRYP_ContextSave->CRYP_CSGCMR[0] = CRYP->CSGCM0R;
567 CRYP_ContextSave->CRYP_CSGCMR[1] = CRYP->CSGCM1R;
568 CRYP_ContextSave->CRYP_CSGCMR[2] = CRYP->CSGCM2R;
569 CRYP_ContextSave->CRYP_CSGCMR[3] = CRYP->CSGCM3R;
570 CRYP_ContextSave->CRYP_CSGCMR[4] = CRYP->CSGCM4R;
571 CRYP_ContextSave->CRYP_CSGCMR[5] = CRYP->CSGCM5R;
572 CRYP_ContextSave->CRYP_CSGCMR[6] = CRYP->CSGCM6R;
573 CRYP_ContextSave->CRYP_CSGCMR[7] = CRYP->CSGCM7R;
574
575 /* When needed, save the DMA status (pointers for IN and OUT messages,
576 number of remaining bytes, etc.) */
577
578 status = SUCCESS;
579 }
580
581 return status;
