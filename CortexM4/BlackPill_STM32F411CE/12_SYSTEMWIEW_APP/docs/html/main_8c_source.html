191void Task_LED(void){
192 //SEGGER_SYSVIEW_RecordVoid(33);//SEGGER_SYSVIEW_MarkStart(0x03);
193 static enum {
194 I_LED_ON=0,
195 S_LED_ON,
196 I_LED_OFF,
197 S_LED_OFF
198 } state = I_LED_ON;
199
200 static clock_t t0, t1;
201
202 t1 = clock();
203
204 switch (state) {
205 case I_LED_ON:
206 SEGGER_SYSVIEW_RecordVoid(50 + I_LED_ON);
207 IO_Write(IOP_LED, 0); // LED on
208
209 t0 = t1;
210 state = S_LED_ON;
211 SEGGER_SYSVIEW_RecordEndCall(50 + I_LED_ON);
212 //break;
213 case S_LED_ON:
214 SEGGER_SYSVIEW_RecordVoid(50 + S_LED_ON);
215 if (t1 - t0 >= CLOCKS_PER_SEC / 10) /*100ms ON*/
216 state = I_LED_OFF;
217 SEGGER_SYSVIEW_RecordEndCall(50 + S_LED_ON);
218 break;
219 case I_LED_OFF:
220 SEGGER_SYSVIEW_RecordVoid(50 + I_LED_OFF);
221 IO_Write(IOP_LED, 1); // LED off
222
223 t0 = t1;
224 state = S_LED_OFF;
225 SEGGER_SYSVIEW_RecordEndCall(50 + I_LED_OFF);
226 //break;
227 case S_LED_OFF:
228 SEGGER_SYSVIEW_RecordVoid(50 + S_LED_OFF);
229 if (t1 - t0 >= 9 * CLOCKS_PER_SEC / 10) /*900ms OFF*/
230 state = I_LED_ON;
231 SEGGER_SYSVIEW_RecordEndCall(50 + S_LED_OFF);
232 break;
233 }
234 //SEGGER_SYSVIEW_RecordEndCall(33);//SEGGER_SYSVIEW_MarkStop(0x03);
235}
