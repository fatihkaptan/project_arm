26void Timer_Init(int tmNo, unsigned prescale, unsigned period, unsigned repeat)
27{
28 TIM_TimeBaseInitTypeDef tmInit;
29 //1) cevresele base clock(CNT) tanimlanir.
30 if(tmNo ==TIMER_1)
31 RCC_APB2PeriphClockCmd(TimRccTab[tmNo],ENABLE);
32 else
33 RCC_APB1PeriphClockCmd(TimRccTab[tmNo],ENABLE);
34
35 //2) timer parametreleri ayarlanir.
36 tmInit.TIM_ClockDivision = TIM_CKD_DIV1;
37 tmInit.TIM_CounterMode = TIM_CounterMode_Up;
38 tmInit.TIM_Period = period - 1;
39 if(period==1)
40 tmInit.TIM_Period = 1;
41 tmInit.TIM_Prescaler = prescale - 1;
42 tmInit.TIM_RepetitionCounter =repeat-1 ;
43
44 //3)ilgili timer modudlu icin ayarlar yuklenir.
45 TIM_TimeBaseInit(TimerTable[tmNo], &tmInit);
46 Timer_Reset(tmNo);
47}

183void TIM3_IRQHandler(void)
184{
185 static uint16_t c0, c1, c2, c3;
186 static uint8_t edge_flag = 0; // 0 = Yükselen, 1 = Düsen
187 // Overflow Interrupt (Timer tasmasi)
188 if (TIM_GetITStatus(TimerTable[TIMER_3], TIM_IT_Update) == SET){
189 SEGGER_SYSVIEW_RecordEnterTimer(3);
190 g_T3Count++;
191 SEGGER_SYSVIEW_RecordExitTimer();
192 TIM_ClearITPendingBit(TimerTable[TIMER_3], TIM_IT_Update);
193 }
194 // Kanal 2 - Periyot Ölçümü (Düsen Kenar)
195 else if (TIM_GetITStatus(TimerTable[TIMER_3], TIM_IT_CC2) == SET){
196 c1 = TIM_GetCapture2(TIM3);
197 // Overflow durumunda düzeltme yap
198 if (c1 >= c0)
199 g_T3CaptureCount = c1 - c0;
200 else
201 g_T3CaptureCount = (0xFFFF - c0) + c1;
202
203 c0 = c1; // Son degeri güncelle
204 TIM_ClearITPendingBit(TimerTable[TIMER_3], TIM_IT_CC2);
205 }
206 // Kanal 1 - Ontime Ölçümü (Both Edge)
207 else if (TIM_GetITStatus(TimerTable[TIMER_3], TIM_IT_CC1) == SET){
208 c3 = TIM_GetCapture1(TIM3);
209 if (edge_flag == 1) // Eger gelen kenar dusen ise
210 {
211 c2 = c3; // Baslangiç zamanini kaydet
212 edge_flag = 0; // Siradaki kenar yukselen olacak
213 }
214 else // Gelen kenar düsen ise
215 {
216 // Overflow durumunda düzeltme yap
217 if (c3 >= c2)
218 g_T3CaptureCountOntime = c3 - c2;
219 else
220 g_T3CaptureCountOntime = (0xFFFF - c2) + c3;
221
222 edge_flag = 1; // Sonraki kenar tekrar dusen olacak
223 }
224 TIM_ClearITPendingBit(TimerTable[TIMER_3], TIM_IT_CC1);
225 }
226}
