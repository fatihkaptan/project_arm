147void WWDG_SetPrescaler(uint32_t WWDG_Prescaler)
148{
149 uint32_t tmpreg = 0;
150 /* Check the parameters */
151 assert_param(IS_WWDG_PRESCALER(WWDG_Prescaler));
152 /* Clear WDGTB[1:0] bits */
153 tmpreg = WWDG->CFR & CFR_WDGTB_MASK;
154 /* Set WDGTB[1:0] bits according to WWDG_Prescaler value */
155 tmpreg |= WWDG_Prescaler;
156 /* Store the new value */
157 WWDG->CFR = tmpreg;
158}
