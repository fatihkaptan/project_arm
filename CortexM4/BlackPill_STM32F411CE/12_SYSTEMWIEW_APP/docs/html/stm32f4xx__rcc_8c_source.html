1310void RCC_GetClocksFreq(RCC_ClocksTypeDef* RCC_Clocks)
1311{
1312 uint32_t tmp = 0, presc = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
1313#if defined(STM32F412xG) || defined(STM32F413_423xx) || defined(STM32F446xx)
1314 uint32_t pllr = 2;
1315#endif /* STM32F412xG || STM32F413_423xx || STM32F446xx */
1316
1317 /* Get SYSCLK source -------------------------------------------------------*/
1318 tmp = RCC->CFGR & RCC_CFGR_SWS;
1319
1320 switch (tmp)
1321 {
1322 case 0x00: /* HSI used as system clock source */
1323 RCC_Clocks->SYSCLK_Frequency = HSI_VALUE;
1324 break;
1325 case 0x04: /* HSE used as system clock source */
1326 RCC_Clocks->SYSCLK_Frequency = HSE_VALUE;
1327 break;
1328 case 0x08: /* PLL P used as system clock source */
1329
1330 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLLM) * PLLN
1331 SYSCLK = PLL_VCO / PLLP
1332 */
1333 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
1334 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
1335
1336 if (pllsource != 0)
1337 {
1338 /* HSE used as PLL clock source */
1339 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
1340 }
1341 else
1342 {
1343 /* HSI used as PLL clock source */
1344 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
1345 }
1346
1347 pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;
1348 RCC_Clocks->SYSCLK_Frequency = pllvco/pllp;
1349 break;
1350
1351#if defined(STM32F412xG) || defined(STM32F413_423xx) || defined(STM32F446xx)
1352 case 0x0C: /* PLL R used as system clock source */
1353 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLLM) * PLLN
1354 SYSCLK = PLL_VCO / PLLR
1355 */
1356 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
1357 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
1358
1359 if (pllsource != 0)
1360 {
1361 /* HSE used as PLL clock source */
1362 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
1363 }
1364 else
1365 {
1366 /* HSI used as PLL clock source */
1367 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
1368 }
1369
1370 pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >>28) + 1 ) *2;
1371 RCC_Clocks->SYSCLK_Frequency = pllvco/pllr;
1372 break;
1373#endif /* STM32F412xG || STM32F413_423xx || STM32F446xx */
1374
1375 default:
1376 RCC_Clocks->SYSCLK_Frequency = HSI_VALUE;
1377 break;
1378 }
1379 /* Compute HCLK, PCLK1 and PCLK2 clocks frequencies ------------------------*/
1380
1381 /* Get HCLK prescaler */
1382 tmp = RCC->CFGR & RCC_CFGR_HPRE;
1383 tmp = tmp >> 4;
1384 presc = APBAHBPrescTable[tmp];
1385 /* HCLK clock frequency */
1386 RCC_Clocks->HCLK_Frequency = RCC_Clocks->SYSCLK_Frequency >> presc;
1387
1388 /* Get PCLK1 prescaler */
1389 tmp = RCC->CFGR & RCC_CFGR_PPRE1;
1390 tmp = tmp >> 10;
1391 presc = APBAHBPrescTable[tmp];
1392 /* PCLK1 clock frequency */
1393 RCC_Clocks->PCLK1_Frequency = RCC_Clocks->HCLK_Frequency >> presc;
1394
1395 /* Get PCLK2 prescaler */
1396 tmp = RCC->CFGR & RCC_CFGR_PPRE2;
1397 tmp = tmp >> 13;
1398 presc = APBAHBPrescTable[tmp];
1399 /* PCLK2 clock frequency */
1400 RCC_Clocks->PCLK2_Frequency = RCC_Clocks->HCLK_Frequency >> presc;
1401}
