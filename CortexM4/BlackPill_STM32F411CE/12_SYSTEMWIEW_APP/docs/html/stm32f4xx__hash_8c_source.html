184void HASH_Init(HASH_InitTypeDef* HASH_InitStruct)
185{
186 /* Check the parameters */
187 assert_param(IS_HASH_ALGOSELECTION(HASH_InitStruct->HASH_AlgoSelection));
188 assert_param(IS_HASH_DATATYPE(HASH_InitStruct->HASH_DataType));
189 assert_param(IS_HASH_ALGOMODE(HASH_InitStruct->HASH_AlgoMode));
190
191 /* Configure the Algorithm used, algorithm mode and the datatype */
192 HASH->CR &= ~ (HASH_CR_ALGO | HASH_CR_DATATYPE | HASH_CR_MODE);
193 HASH->CR |= (HASH_InitStruct->HASH_AlgoSelection | \
194 HASH_InitStruct->HASH_DataType | \
195 HASH_InitStruct->HASH_AlgoMode);
196
197 /* if algorithm mode is HMAC, set the Key */
198 if(HASH_InitStruct->HASH_AlgoMode == HASH_AlgoMode_HMAC)
199 {
200 assert_param(IS_HASH_HMAC_KEYTYPE(HASH_InitStruct->HASH_HMACKeyType));
201 HASH->CR &= ~HASH_CR_LKEY;
202 HASH->CR |= HASH_InitStruct->HASH_HMACKeyType;
203 }
204
205 /* Reset the HASH processor core, so that the HASH will be ready to compute
206 the message digest of a new message */
207 HASH->CR |= HASH_CR_INIT;
208}

603FlagStatus HASH_GetFlagStatus(uint32_t HASH_FLAG)
604{
605 FlagStatus bitstatus = RESET;
606 uint32_t tempreg = 0;
607
608 /* Check the parameters */
609 assert_param(IS_HASH_GET_FLAG(HASH_FLAG));
610
611 /* check if the FLAG is in CR register */
612 if ((HASH_FLAG & HASH_FLAG_DINNE) != (uint32_t)RESET )
613 {
614 tempreg = HASH->CR;
615 }
616 else /* The FLAG is in SR register */
617 {
618 tempreg = HASH->SR;
619 }
620
621 /* Check the status of the specified HASH flag */
622 if ((tempreg & HASH_FLAG) != (uint32_t)RESET)
623 {
624 /* HASH is set */
625 bitstatus = SET;
626 }
627 else
628 {
629 /* HASH_FLAG is reset */
630 bitstatus = RESET;
631 }
632
633 /* Return the HASH_FLAG status */
634 return bitstatus;
