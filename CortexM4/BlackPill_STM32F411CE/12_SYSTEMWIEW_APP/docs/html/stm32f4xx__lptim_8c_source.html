161void LPTIM_Init(LPTIM_TypeDef* LPTIMx, LPTIM_InitTypeDef* LPTIM_InitStruct)
162{
163 uint32_t tmpreg1 = 0;
164
165 /* Check the parameters */
166 assert_param(IS_LPTIM_ALL_PERIPH(LPTIMx));
167 assert_param(IS_LPTIM_CLOCK_SOURCE(LPTIM_InitStruct->LPTIM_ClockSource));
168 assert_param(IS_LPTIM_CLOCK_PRESCALER(LPTIM_InitStruct->LPTIM_Prescaler));
169 assert_param(IS_LPTIM_WAVEFORM(LPTIM_InitStruct->LPTIM_Waveform));
170 assert_param(IS_LPTIM_OUTPUT_POLARITY(LPTIM_InitStruct->LPTIM_OutputPolarity));
171
172 /* Get the LPTIMx CFGR value */
173 tmpreg1 = LPTIMx->CFGR;
174
175 /* Clear CKSEL, PRESC, WAVE and WAVEPOL bits */
176 tmpreg1 &= CFGR_INIT_CLEAR_MASK;
177
178 /* Set or Reset CKSEL bit according to LPTIM_ClockSource value */
179 /* Set or Reset PRESC bits according to LPTIM_Prescaler value */
180 /* Set or Reset WAVE bit according to LPTIM_Waveform value */
181 /* Set or Reset WAVEPOL bit according to LPTIM_OutputPolarity value */
182 tmpreg1 |= (LPTIM_InitStruct->LPTIM_ClockSource | LPTIM_InitStruct->LPTIM_Prescaler
183 |LPTIM_InitStruct->LPTIM_Waveform | LPTIM_InitStruct->LPTIM_OutputPolarity);
184
185 /* Write to LPTIMx CFGR */
186 LPTIMx->CFGR = tmpreg1;
187}

900ITStatus LPTIM_GetITStatus(LPTIM_TypeDef* LPTIMx, uint32_t LPTIM_IT)
901{
902 ITStatus bitstatus = RESET;
903 uint32_t itstatus = 0x0, itenable = 0x0;
904
905 /* Check the parameters */
906 assert_param(IS_LPTIM_ALL_PERIPH(LPTIMx));
907 assert_param(IS_LPTIM_IT(LPTIM_IT));
908
909 /* Get the Interrupt Status bit value */
910 itstatus = LPTIMx->ISR & LPTIM_IT;
911
912 /* Check if the Interrupt is enabled */
913 itenable = LPTIMx->IER & LPTIM_IT;
914
915 if((itstatus != RESET) && (itenable != RESET))
916 {
917 bitstatus = SET;
918 }
919 else
920 {
921 bitstatus = RESET;
922 }
923 return bitstatus;
924}
