97void SPDIFRX_Init(SPDIFRX_InitTypeDef* SPDIFRX_InitStruct)
98{
99 uint32_t tmpreg = 0;
100
101 /* Check the SPDIFRX parameters */
102 assert_param(IS_STEREO_MODE(SPDIFRX_InitStruct->SPDIFRX_StereoMode));
103 assert_param(IS_SPDIFRX_INPUT_SELECT(SPDIFRX_InitStruct->SPDIFRX_InputSelection));
104 assert_param(IS_SPDIFRX_MAX_RETRIES(SPDIFRX_InitStruct->SPDIFRX_Retries));
105 assert_param(IS_SPDIFRX_WAIT_FOR_ACTIVITY(SPDIFRX_InitStruct->SPDIFRX_WaitForActivity));
106 assert_param(IS_SPDIFRX_CHANNEL(SPDIFRX_InitStruct->SPDIFRX_ChannelSelection));
107 assert_param(IS_SPDIFRX_DATA_FORMAT(SPDIFRX_InitStruct->SPDIFRX_DataFormat));
108
109 /* SPDIFRX CR Configuration */
110 /* Get the SPDIFRX CR value */
111 tmpreg = SPDIFRX->CR;
112 /* Clear INSEL, WFA, NBTR, CHSEL, DRFMT and RXSTEO bits */
113 tmpreg &= CR_CLEAR_MASK;
114 /* Configure SPDIFRX: Input selection, Maximum allowed re-tries during synchronization phase,
115 wait for activity, Channel Selection, Data samples format and stereo/mono mode */
116 /* Set INSEL bits according to SPDIFRX_InputSelection value */
117 /* Set WFA bit according to SPDIFRX_WaitForActivity value */
118 /* Set NBTR bit according to SPDIFRX_Retries value */
119 /* Set CHSEL bit according to SPDIFRX_ChannelSelection value */
120 /* Set DRFMT bits according to SPDIFRX_DataFormat value */
121 /* Set RXSTEO bit according to SPDIFRX_StereoMode value */
122
123 tmpreg |= (uint32_t)(SPDIFRX_InitStruct->SPDIFRX_InputSelection | SPDIFRX_InitStruct->SPDIFRX_WaitForActivity |
124 SPDIFRX_InitStruct->SPDIFRX_Retries | SPDIFRX_InitStruct->SPDIFRX_ChannelSelection |
125 SPDIFRX_InitStruct->SPDIFRX_DataFormat | SPDIFRX_InitStruct->SPDIFRX_StereoMode
126 );
127
128 /* Write to SPDIFRX CR */
129 SPDIFRX->CR = tmpreg;
130}
