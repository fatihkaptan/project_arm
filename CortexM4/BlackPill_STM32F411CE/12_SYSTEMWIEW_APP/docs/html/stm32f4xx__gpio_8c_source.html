195void GPIO_Init(GPIO_TypeDef* GPIOx, GPIO_InitTypeDef* GPIO_InitStruct)
196{
197 uint32_t pinpos = 0x00, pos = 0x00 , currentpin = 0x00;
198
199 /* Check the parameters */
200 assert_param(IS_GPIO_ALL_PERIPH(GPIOx));
201 assert_param(IS_GPIO_PIN(GPIO_InitStruct->GPIO_Pin));
202 assert_param(IS_GPIO_MODE(GPIO_InitStruct->GPIO_Mode));
203 assert_param(IS_GPIO_PUPD(GPIO_InitStruct->GPIO_PuPd));
204
205 /* ------------------------- Configure the port pins ---------------- */
206 /*-- GPIO Mode Configuration --*/
207 for (pinpos = 0x00; pinpos < 0x10; pinpos++)
208 {
209 pos = ((uint32_t)0x01) << pinpos;
210 /* Get the port pins position */
211 currentpin = (GPIO_InitStruct->GPIO_Pin) & pos;
212
213 if (currentpin == pos)
214 {
215 GPIOx->MODER &= ~(GPIO_MODER_MODER0 << (pinpos * 2));
216 GPIOx->MODER |= (((uint32_t)GPIO_InitStruct->GPIO_Mode) << (pinpos * 2));
217
218 if ((GPIO_InitStruct->GPIO_Mode == GPIO_Mode_OUT) || (GPIO_InitStruct->GPIO_Mode == GPIO_Mode_AF))
219 {
220 /* Check Speed mode parameters */
221 assert_param(IS_GPIO_SPEED(GPIO_InitStruct->GPIO_Speed));
222
223 /* Speed mode configuration */
224 GPIOx->OSPEEDR &= ~(GPIO_OSPEEDER_OSPEEDR0 << (pinpos * 2));
225 GPIOx->OSPEEDR |= ((uint32_t)(GPIO_InitStruct->GPIO_Speed) << (pinpos * 2));
226
227 /* Check Output mode parameters */
228 assert_param(IS_GPIO_OTYPE(GPIO_InitStruct->GPIO_OType));
229
230 /* Output mode configuration*/
231 GPIOx->OTYPER &= ~((GPIO_OTYPER_OT_0) << ((uint16_t)pinpos)) ;
232 GPIOx->OTYPER |= (uint16_t)(((uint16_t)GPIO_InitStruct->GPIO_OType) << ((uint16_t)pinpos));
233 }
234
235 /* Pull-up Pull down resistor configuration*/
236 GPIOx->PUPDR &= ~(GPIO_PUPDR_PUPDR0 << ((uint16_t)pinpos * 2));
237 GPIOx->PUPDR |= (((uint32_t)GPIO_InitStruct->GPIO_PuPd) << (pinpos * 2));
238 }
239 }
