173void I2C_Init(I2C_TypeDef* I2Cx, I2C_InitTypeDef* I2C_InitStruct)
174{
175 uint16_t tmpreg = 0, freqrange = 0;
176 uint16_t result = 0x04;
177 uint32_t pclk1 = 8000000;
178 RCC_ClocksTypeDef rcc_clocks;
179 /* Check the parameters */
180 assert_param(IS_I2C_ALL_PERIPH(I2Cx));
181 assert_param(IS_I2C_CLOCK_SPEED(I2C_InitStruct->I2C_ClockSpeed));
182 assert_param(IS_I2C_MODE(I2C_InitStruct->I2C_Mode));
183 assert_param(IS_I2C_DUTY_CYCLE(I2C_InitStruct->I2C_DutyCycle));
184 assert_param(IS_I2C_OWN_ADDRESS1(I2C_InitStruct->I2C_OwnAddress1));
185 assert_param(IS_I2C_ACK_STATE(I2C_InitStruct->I2C_Ack));
186 assert_param(IS_I2C_ACKNOWLEDGE_ADDRESS(I2C_InitStruct->I2C_AcknowledgedAddress));
187
188/*---------------------------- I2Cx CR2 Configuration ------------------------*/
189 /* Get the I2Cx CR2 value */
190 tmpreg = I2Cx->CR2;
191 /* Clear frequency FREQ[5:0] bits */
192 tmpreg &= (uint16_t)~((uint16_t)I2C_CR2_FREQ);
193 /* Get pclk1 frequency value */
194 RCC_GetClocksFreq(&rcc_clocks);
195 pclk1 = rcc_clocks.PCLK1_Frequency;
196 /* Set frequency bits depending on pclk1 value */
197 freqrange = (uint16_t)(pclk1 / 1000000);
198 tmpreg |= freqrange;
199 /* Write to I2Cx CR2 */
200 I2Cx->CR2 = tmpreg;
201
202/*---------------------------- I2Cx CCR Configuration ------------------------*/
203 /* Disable the selected I2C peripheral to configure TRISE */
204 I2Cx->CR1 &= (uint16_t)~((uint16_t)I2C_CR1_PE);
205 /* Reset tmpreg value */
206 /* Clear F/S, DUTY and CCR[11:0] bits */
207 tmpreg = 0;
208
209 /* Configure speed in standard mode */
210 if (I2C_InitStruct->I2C_ClockSpeed <= 100000)
211 {
212 /* Standard mode speed calculate */
213 result = (uint16_t)(pclk1 / (I2C_InitStruct->I2C_ClockSpeed << 1));
214 /* Test if CCR value is under 0x4*/
215 if (result < 0x04)
216 {
217 /* Set minimum allowed value */
218 result = 0x04;
219 }
220 /* Set speed value for standard mode */
221 tmpreg |= result;
222 /* Set Maximum Rise Time for standard mode */
223 I2Cx->TRISE = freqrange + 1;
224 }
225 /* Configure speed in fast mode */
226 /* To use the I2C at 400 KHz (in fast mode), the PCLK1 frequency (I2C peripheral
227 input clock) must be a multiple of 10 MHz */
228 else /*(I2C_InitStruct->I2C_ClockSpeed <= 400000)*/
229 {
230 if (I2C_InitStruct->I2C_DutyCycle == I2C_DutyCycle_2)
231 {
232 /* Fast mode speed calculate: Tlow/Thigh = 2 */
233 result = (uint16_t)(pclk1 / (I2C_InitStruct->I2C_ClockSpeed * 3));
234 }
235 else /*I2C_InitStruct->I2C_DutyCycle == I2C_DutyCycle_16_9*/
236 {
237 /* Fast mode speed calculate: Tlow/Thigh = 16/9 */
238 result = (uint16_t)(pclk1 / (I2C_InitStruct->I2C_ClockSpeed * 25));
239 /* Set DUTY bit */
240 result |= I2C_DutyCycle_16_9;
241 }
242
243 /* Test if CCR value is under 0x1*/
244 if ((result & I2C_CCR_CCR) == 0)
245 {
246 /* Set minimum allowed value */
247 result |= (uint16_t)0x0001;
248 }
249 /* Set speed value and set F/S bit for fast mode */
250 tmpreg |= (uint16_t)(result | I2C_CCR_FS);
251 /* Set Maximum Rise Time for fast mode */
252 I2Cx->TRISE = (uint16_t)(((freqrange * (uint16_t)300) / (uint16_t)1000) + (uint16_t)1);
253 }
254
255 /* Write to I2Cx CCR */
256 I2Cx->CCR = tmpreg;
257 /* Enable the selected I2C peripheral */
258 I2Cx->CR1 |= I2C_CR1_PE;
259
260/*---------------------------- I2Cx CR1 Configuration ------------------------*/
261 /* Get the I2Cx CR1 value */
262 tmpreg = I2Cx->CR1;
263 /* Clear ACK, SMBTYPE and SMBUS bits */
264 tmpreg &= CR1_CLEAR_MASK;
265 /* Configure I2Cx: mode and acknowledgement */
266 /* Set SMBTYPE and SMBUS bits according to I2C_Mode value */
267 /* Set ACK bit according to I2C_Ack value */
268 tmpreg |= (uint16_t)((uint32_t)I2C_InitStruct->I2C_Mode | I2C_InitStruct->I2C_Ack);
269 /* Write to I2Cx CR1 */
270 I2Cx->CR1 = tmpreg;
271
272/*---------------------------- I2Cx OAR1 Configuration -----------------------*/
273 /* Set I2Cx Own Address1 and acknowledged address */
274 I2Cx->OAR1 = (I2C_InitStruct->I2C_AcknowledgedAddress | I2C_InitStruct->I2C_OwnAddress1);
275}

1151ErrorStatus I2C_CheckEvent(I2C_TypeDef* I2Cx, uint32_t I2C_EVENT)
1152{
1153 uint32_t lastevent = 0;
1154 uint32_t flag1 = 0, flag2 = 0;
1155 ErrorStatus status = ERROR;
1156
1157 /* Check the parameters */
1158 assert_param(IS_I2C_ALL_PERIPH(I2Cx));
1159 assert_param(IS_I2C_EVENT(I2C_EVENT));
1160
1161 /* Read the I2Cx status register */
1162 flag1 = I2Cx->SR1;
1163
1164 /* I2C_SR2 must be read only when ADDR is found set in I2C_SR1 or when the STOPF bit is cleared */
1165 if((flag1 & I2C_SR1_ADDR) || (flag1 & ~I2C_SR1_STOPF))
1166 {
1167 flag2 = I2Cx->SR2;
1168 }
1169 else
1170 {
1171 return ERROR;
1172 }
1173 flag2 = flag2 << 16;
1174
1175 /* Get the last event value from I2C status register */
1176 lastevent = (flag1 | flag2) & FLAG_MASK;
1177
1178 /* Check whether the last event contains the I2C_EVENT */
1179 if ((lastevent & I2C_EVENT) == I2C_EVENT)
1180 {
1181 /* SUCCESS: last event is equal to I2C_EVENT */
1182 status = SUCCESS;
1183 }
1184 else
1185 {
1186 /* ERROR: last event is different from I2C_EVENT */
1187 status = ERROR;
1188 }
1189 /* Return status */
1190 return status;
