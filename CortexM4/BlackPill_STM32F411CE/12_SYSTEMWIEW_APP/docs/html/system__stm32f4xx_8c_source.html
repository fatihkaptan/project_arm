476void SystemInit(void)
477{
478 /* FPU settings ------------------------------------------------------------*/
479 #if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
480 SCB->CPACR |= ((3UL << 10*2)|(3UL << 11*2)); /* set CP10 and CP11 Full Access */
481 #endif
482 /* Reset the RCC clock configuration to the default reset state ------------*/
483 /* Set HSION bit */
484 RCC->CR |= (uint32_t)0x00000001;
485
486 /* Reset CFGR register */
487 RCC->CFGR = 0x00000000;
488
489 /* Reset HSEON, CSSON and PLLON bits */
490 RCC->CR &= (uint32_t)0xFEF6FFFF;
491
492 /* Reset PLLCFGR register */
493 RCC->PLLCFGR = 0x24003010;
494
495 /* Reset HSEBYP bit */
496 RCC->CR &= (uint32_t)0xFFFBFFFF;
497
498 /* Disable all interrupts */
499 RCC->CIR = 0x00000000;
500
501#if defined(DATA_IN_ExtSRAM) || defined(DATA_IN_ExtSDRAM)
502 SystemInit_ExtMemCtl();
503#endif /* DATA_IN_ExtSRAM || DATA_IN_ExtSDRAM */
504
505 /* Configure the System clock source, PLL Multiplier and Divider factors,
506 AHB/APBx prescalers and Flash settings ----------------------------------*/
507 SetSysClock();
508
509 /* Configure the Vector Table location add offset address ------------------*/
510#ifdef VECT_TAB_SRAM
511 SCB->VTOR = SRAM_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal SRAM */
512#else
513 SCB->VTOR = FLASH_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal FLASH */
514#endif
515}

553void SystemCoreClockUpdate(void)
554{
555 uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
556#if defined(STM32F412xG) || defined(STM32F413_423xx) || defined(STM32F446xx)
557 uint32_t pllr = 2;
558#endif /* STM32F412xG || STM32F413_423xx || STM32F446xx */
559 /* Get SYSCLK source -------------------------------------------------------*/
560 tmp = RCC->CFGR & RCC_CFGR_SWS;
561
562 switch (tmp)
563 {
564 case 0x00: /* HSI used as system clock source */
565 SystemCoreClock = HSI_VALUE;
566 break;
567 case 0x04: /* HSE used as system clock source */
568 SystemCoreClock = HSE_VALUE;
569 break;
570 case 0x08: /* PLL P used as system clock source */
571 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
572 SYSCLK = PLL_VCO / PLL_P
573 */
574 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
575 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
576
577#if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F401xx) || defined(STM32F412xG) || defined(STM32F413_423xx) || defined(STM32F446xx) || defined(STM32F469_479xx)
578 if (pllsource != 0)
579 {
580 /* HSE used as PLL clock source */
581 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
582 }
583 else
584 {
585 /* HSI used as PLL clock source */
586 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
587 }
588#elif defined(STM32F410xx) || defined(STM32F411xE)
589#if defined(USE_HSE_BYPASS)
590 if (pllsource != 0)
591 {
592 /* HSE used as PLL clock source */
593 pllvco = (HSE_BYPASS_INPUT_FREQUENCY / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
594 }
595#else
596 if (pllsource == 0)
597 {
598 /* HSI used as PLL clock source */
599 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
600 }
601#endif /* USE_HSE_BYPASS */
602#endif /* STM32F40_41xxx || STM32F427_437xx || STM32F429_439xx || STM32F401xx || STM32F412xG || STM32F413_423xx || STM32F446xx || STM32F469_479xx */
603 pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;
604 SystemCoreClock = pllvco/pllp;
605 break;
606#if defined(STM32F412xG) || defined(STM32F413_423xx) || defined(STM32F446xx)
607 case 0x0C: /* PLL R used as system clock source */
608 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
609 SYSCLK = PLL_VCO / PLL_R
610 */
611 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
612 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
613 if (pllsource != 0)
614 {
615 /* HSE used as PLL clock source */
616 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
617 }
618 else
619 {
620 /* HSI used as PLL clock source */
621 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
622 }
623
624 pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >>28) + 1 ) *2;
625 SystemCoreClock = pllvco/pllr;
626 break;
627#endif /* STM32F412xG || STM32F413_423xx || STM32F446xx */
628 default:
629 SystemCoreClock = HSI_VALUE;
630 break;
631 }
632 /* Compute HCLK frequency --------------------------------------------------*/
633 /* Get HCLK prescaler */
634 tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
635 /* HCLK frequency */
636 SystemCoreClock >>= tmp;
637}
