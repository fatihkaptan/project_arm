277void SPI_Init(SPI_TypeDef* SPIx, SPI_InitTypeDef* SPI_InitStruct)
278{
279 uint16_t tmpreg = 0;
280
281 /* check the parameters */
282 assert_param(IS_SPI_ALL_PERIPH(SPIx));
283
284 /* Check the SPI parameters */
285 assert_param(IS_SPI_DIRECTION_MODE(SPI_InitStruct->SPI_Direction));
286 assert_param(IS_SPI_MODE(SPI_InitStruct->SPI_Mode));
287 assert_param(IS_SPI_DATASIZE(SPI_InitStruct->SPI_DataSize));
288 assert_param(IS_SPI_CPOL(SPI_InitStruct->SPI_CPOL));
289 assert_param(IS_SPI_CPHA(SPI_InitStruct->SPI_CPHA));
290 assert_param(IS_SPI_NSS(SPI_InitStruct->SPI_NSS));
291 assert_param(IS_SPI_BAUDRATE_PRESCALER(SPI_InitStruct->SPI_BaudRatePrescaler));
292 assert_param(IS_SPI_FIRST_BIT(SPI_InitStruct->SPI_FirstBit));
293 assert_param(IS_SPI_CRC_POLYNOMIAL(SPI_InitStruct->SPI_CRCPolynomial));
294
295/*---------------------------- SPIx CR1 Configuration ------------------------*/
296 /* Get the SPIx CR1 value */
297 tmpreg = SPIx->CR1;
298 /* Clear BIDIMode, BIDIOE, RxONLY, SSM, SSI, LSBFirst, BR, MSTR, CPOL and CPHA bits */
299 tmpreg &= CR1_CLEAR_MASK;
300 /* Configure SPIx: direction, NSS management, first transmitted bit, BaudRate prescaler
301 master/salve mode, CPOL and CPHA */
302 /* Set BIDImode, BIDIOE and RxONLY bits according to SPI_Direction value */
303 /* Set SSM, SSI and MSTR bits according to SPI_Mode and SPI_NSS values */
304 /* Set LSBFirst bit according to SPI_FirstBit value */
305 /* Set BR bits according to SPI_BaudRatePrescaler value */
306 /* Set CPOL bit according to SPI_CPOL value */
307 /* Set CPHA bit according to SPI_CPHA value */
308 tmpreg |= (uint16_t)((uint32_t)SPI_InitStruct->SPI_Direction | SPI_InitStruct->SPI_Mode |
309 SPI_InitStruct->SPI_DataSize | SPI_InitStruct->SPI_CPOL |
310 SPI_InitStruct->SPI_CPHA | SPI_InitStruct->SPI_NSS |
311 SPI_InitStruct->SPI_BaudRatePrescaler | SPI_InitStruct->SPI_FirstBit);
312 /* Write to SPIx CR1 */
313 SPIx->CR1 = tmpreg;
314
315 /* Activate the SPI mode (Reset I2SMOD bit in I2SCFGR register) */
316 SPIx->I2SCFGR &= (uint16_t)~((uint16_t)SPI_I2SCFGR_I2SMOD);
317/*---------------------------- SPIx CRCPOLY Configuration --------------------*/
318 /* Write to SPIx CRCPOLY */
319 SPIx->CRCPR = SPI_InitStruct->SPI_CRCPolynomial;
320}

341void I2S_Init(SPI_TypeDef* SPIx, I2S_InitTypeDef* I2S_InitStruct)
342{
343 uint16_t tmpreg = 0, i2sdiv = 2, i2sodd = 0, packetlength = 1;
344 uint32_t tmp = 0, i2sclk = 0;
345#ifndef I2S_EXTERNAL_CLOCK_VAL
346 uint32_t pllm = 0, plln = 0, pllr = 0;
347#endif /* I2S_EXTERNAL_CLOCK_VAL */
348
349 /* Check the I2S parameters */
350 assert_param(IS_SPI_23_PERIPH(SPIx));
351 assert_param(IS_I2S_MODE(I2S_InitStruct->I2S_Mode));
352 assert_param(IS_I2S_STANDARD(I2S_InitStruct->I2S_Standard));
353 assert_param(IS_I2S_DATA_FORMAT(I2S_InitStruct->I2S_DataFormat));
354 assert_param(IS_I2S_MCLK_OUTPUT(I2S_InitStruct->I2S_MCLKOutput));
355 assert_param(IS_I2S_AUDIO_FREQ(I2S_InitStruct->I2S_AudioFreq));
356 assert_param(IS_I2S_CPOL(I2S_InitStruct->I2S_CPOL));
357
358/*----------------------- SPIx I2SCFGR & I2SPR Configuration -----------------*/
359 /* Clear I2SMOD, I2SE, I2SCFG, PCMSYNC, I2SSTD, CKPOL, DATLEN and CHLEN bits */
360 SPIx->I2SCFGR &= I2SCFGR_CLEAR_MASK;
361 SPIx->I2SPR = 0x0002;
362
363 /* Get the I2SCFGR register value */
364 tmpreg = SPIx->I2SCFGR;
365
366 /* If the default value has to be written, reinitialize i2sdiv and i2sodd*/
367 if(I2S_InitStruct->I2S_AudioFreq == I2S_AudioFreq_Default)
368 {
369 i2sodd = (uint16_t)0;
370 i2sdiv = (uint16_t)2;
371 }
372 /* If the requested audio frequency is not the default, compute the prescaler */
373 else
374 {
375 /* Check the frame length (For the Prescaler computing) *******************/
376 if(I2S_InitStruct->I2S_DataFormat == I2S_DataFormat_16b)
377 {
378 /* Packet length is 16 bits */
379 packetlength = 16;
380 }
381 else
382 {
383 /* Packet length is 32 bits */
384 packetlength = 32;
385 }
386
387 if(I2S_InitStruct->I2S_Standard <= I2S_Standard_LSB)
388 {
389 /* In I2S standard packet length is multiplied by 2 */
390 packetlength = packetlength * 2;
391 }
392
393 /* Get I2S source Clock frequency ****************************************/
394
395 /* If an external I2S clock has to be used, this define should be set
396 in the project configuration or in the stm32f4xx_conf.h file */
397 #ifdef I2S_EXTERNAL_CLOCK_VAL
398 /* Set external clock as I2S clock source */
399 if ((RCC->CFGR & RCC_CFGR_I2SSRC) == 0)
400 {
401 RCC->CFGR |= (uint32_t)RCC_CFGR_I2SSRC;
402 }
403
404 /* Set the I2S clock to the external clock value */
405 i2sclk = I2S_EXTERNAL_CLOCK_VAL;
406
407 #else /* There is no define for External I2S clock source */
408 /* Set PLLI2S as I2S clock source */
409 if ((RCC->CFGR & RCC_CFGR_I2SSRC) != 0)
410 {
411 RCC->CFGR &= ~(uint32_t)RCC_CFGR_I2SSRC;
412 }
413
414 /* Get the PLLI2SN value */
415 plln = (uint32_t)(((RCC->PLLI2SCFGR & RCC_PLLI2SCFGR_PLLI2SN) >> 6) & \
416 (RCC_PLLI2SCFGR_PLLI2SN >> 6));
417
418 /* Get the PLLI2SR value */
419 pllr = (uint32_t)(((RCC->PLLI2SCFGR & RCC_PLLI2SCFGR_PLLI2SR) >> 28) & \
420 (RCC_PLLI2SCFGR_PLLI2SR >> 28));
421
422 /* Get the PLLM value */
423 pllm = (uint32_t)(RCC->PLLCFGR & RCC_PLLCFGR_PLLM);
424
425 if((RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) == RCC_PLLCFGR_PLLSRC_HSE)
426 {
427 /* Get the I2S source clock value */
428 i2sclk = (uint32_t)(((HSE_VALUE / pllm) * plln) / pllr);
429 }
430 else
431 { /* Get the I2S source clock value */
432 i2sclk = (uint32_t)(((HSI_VALUE / pllm) * plln) / pllr);
433 }
434 #endif /* I2S_EXTERNAL_CLOCK_VAL */
435
436 /* Compute the Real divider depending on the MCLK output state, with a floating point */
437 if(I2S_InitStruct->I2S_MCLKOutput == I2S_MCLKOutput_Enable)
438 {
439 /* MCLK output is enabled */
440 tmp = (uint16_t)(((((i2sclk / 256) * 10) / I2S_InitStruct->I2S_AudioFreq)) + 5);
441 }
442 else
443 {
444 /* MCLK output is disabled */
445 tmp = (uint16_t)(((((i2sclk / (32 * packetlength)) *10 ) / I2S_InitStruct->I2S_AudioFreq)) + 5);
446 }
447
448 /* Remove the flatting point */
449 tmp = tmp / 10;
450
451 /* Check the parity of the divider */
452 i2sodd = (uint16_t)(tmp & (uint16_t)0x0001);
453
454 /* Compute the i2sdiv prescaler */
455 i2sdiv = (uint16_t)((tmp - i2sodd) / 2);
456
457 /* Get the Mask for the Odd bit (SPI_I2SPR[8]) register */
458 i2sodd = (uint16_t) (i2sodd << 8);
459 }
460
461 /* Test if the divider is 1 or 0 or greater than 0xFF */
462 if ((i2sdiv < 2) || (i2sdiv > 0xFF))
463 {
464 /* Set the default values */
465 i2sdiv = 2;
466 i2sodd = 0;
467 }
468
469 /* Write to SPIx I2SPR register the computed value */
470 SPIx->I2SPR = (uint16_t)((uint16_t)i2sdiv | (uint16_t)(i2sodd | (uint16_t)I2S_InitStruct->I2S_MCLKOutput));
471
472 /* Configure the I2S with the SPI_InitStruct values */
473 tmpreg |= (uint16_t)((uint16_t)SPI_I2SCFGR_I2SMOD | (uint16_t)(I2S_InitStruct->I2S_Mode | \
474 (uint16_t)(I2S_InitStruct->I2S_Standard | (uint16_t)(I2S_InitStruct->I2S_DataFormat | \
475 (uint16_t)I2S_InitStruct->I2S_CPOL))));
476
477#if defined(SPI_I2SCFGR_ASTRTEN)
478 if((I2S_InitStruct->I2S_Standard == I2S_Standard_PCMShort) || (I2S_InitStruct->I2S_Standard == I2S_Standard_PCMLong))
479 {
480 /* Write to SPIx I2SCFGR */
481 SPIx->I2SCFGR = tmpreg | SPI_I2SCFGR_ASTRTEN;
482 }
483#else
484 /* Write to SPIx I2SCFGR */
485 SPIx->I2SCFGR = tmpreg ;
486#endif
487}
