152void CEC_Init(CEC_InitTypeDef* CEC_InitStruct)
153{
154 uint32_t tmpreg = 0;
155
156 /* Check the parameters */
157 assert_param(IS_CEC_SIGNAL_FREE_TIME(CEC_InitStruct->CEC_SignalFreeTime));
158 assert_param(IS_CEC_RX_TOLERANCE(CEC_InitStruct->CEC_RxTolerance));
159 assert_param(IS_CEC_STOP_RECEPTION(CEC_InitStruct->CEC_StopReception));
160 assert_param(IS_CEC_BIT_RISING_ERROR(CEC_InitStruct->CEC_BitRisingError));
161 assert_param(IS_CEC_LONG_BIT_PERIOD_ERROR(CEC_InitStruct->CEC_LongBitPeriodError));
162 assert_param(IS_CEC_BDR_NO_GEN_ERROR(CEC_InitStruct->CEC_BRDNoGen));
163 assert_param(IS_CEC_SFT_OPTION(CEC_InitStruct->CEC_SFTOption));
164
165 /* Get the CEC CFGR value */
166 tmpreg = CEC->CFGR;
167
168 /* Clear CFGR bits */
169 tmpreg &= CFGR_CLEAR_MASK;
170
171 /* Configure the CEC peripheral */
172 tmpreg |= (CEC_InitStruct->CEC_SignalFreeTime | CEC_InitStruct->CEC_RxTolerance |
173 CEC_InitStruct->CEC_StopReception | CEC_InitStruct->CEC_BitRisingError |
174 CEC_InitStruct->CEC_LongBitPeriodError| CEC_InitStruct->CEC_BRDNoGen |
175 CEC_InitStruct->CEC_SFTOption);
176
177 /* Write to CEC CFGR register */
178 CEC->CFGR = tmpreg;
179}
