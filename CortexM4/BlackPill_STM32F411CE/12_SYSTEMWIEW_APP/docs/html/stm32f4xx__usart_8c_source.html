239void USART_Init(USART_TypeDef* USARTx, USART_InitTypeDef* USART_InitStruct)
240{
241 uint32_t tmpreg = 0x00, apbclock = 0x00;
242 uint32_t integerdivider = 0x00;
243 uint32_t fractionaldivider = 0x00;
244 RCC_ClocksTypeDef RCC_ClocksStatus;
245
246 /* Check the parameters */
247 assert_param(IS_USART_ALL_PERIPH(USARTx));
248 assert_param(IS_USART_BAUDRATE(USART_InitStruct->USART_BaudRate));
249 assert_param(IS_USART_WORD_LENGTH(USART_InitStruct->USART_WordLength));
250 assert_param(IS_USART_STOPBITS(USART_InitStruct->USART_StopBits));
251 assert_param(IS_USART_PARITY(USART_InitStruct->USART_Parity));
252 assert_param(IS_USART_MODE(USART_InitStruct->USART_Mode));
253 assert_param(IS_USART_HARDWARE_FLOW_CONTROL(USART_InitStruct->USART_HardwareFlowControl));
254
255 /* The hardware flow control is available only for USART1, USART2, USART3 and USART6 */
256 if (USART_InitStruct->USART_HardwareFlowControl != USART_HardwareFlowControl_None)
257 {
258 assert_param(IS_USART_1236_PERIPH(USARTx));
259 }
260
261/*---------------------------- USART CR2 Configuration -----------------------*/
262 tmpreg = USARTx->CR2;
263
264 /* Clear STOP[13:12] bits */
265 tmpreg &= (uint32_t)~((uint32_t)USART_CR2_STOP);
266
267 /* Configure the USART Stop Bits, Clock, CPOL, CPHA and LastBit :
268 Set STOP[13:12] bits according to USART_StopBits value */
269 tmpreg |= (uint32_t)USART_InitStruct->USART_StopBits;
270
271 /* Write to USART CR2 */
272 USARTx->CR2 = (uint16_t)tmpreg;
273
274/*---------------------------- USART CR1 Configuration -----------------------*/
275 tmpreg = USARTx->CR1;
276
277 /* Clear M, PCE, PS, TE and RE bits */
278 tmpreg &= (uint32_t)~((uint32_t)CR1_CLEAR_MASK);
279
280 /* Configure the USART Word Length, Parity and mode:
281 Set the M bits according to USART_WordLength value
282 Set PCE and PS bits according to USART_Parity value
283 Set TE and RE bits according to USART_Mode value */
284 tmpreg |= (uint32_t)USART_InitStruct->USART_WordLength | USART_InitStruct->USART_Parity |
285 USART_InitStruct->USART_Mode;
286
287 /* Write to USART CR1 */
288 USARTx->CR1 = (uint16_t)tmpreg;
289
290/*---------------------------- USART CR3 Configuration -----------------------*/
291 tmpreg = USARTx->CR3;
292
293 /* Clear CTSE and RTSE bits */
294 tmpreg &= (uint32_t)~((uint32_t)CR3_CLEAR_MASK);
295
296 /* Configure the USART HFC :
297 Set CTSE and RTSE bits according to USART_HardwareFlowControl value */
298 tmpreg |= USART_InitStruct->USART_HardwareFlowControl;
299
300 /* Write to USART CR3 */
301 USARTx->CR3 = (uint16_t)tmpreg;
302
303/*---------------------------- USART BRR Configuration -----------------------*/
304 /* Configure the USART Baud Rate */
305 RCC_GetClocksFreq(&RCC_ClocksStatus);
306
307 if ((USARTx == USART1) || (USARTx == USART6))
308 {
309 apbclock = RCC_ClocksStatus.PCLK2_Frequency;
310 }
311 else
312 {
313 apbclock = RCC_ClocksStatus.PCLK1_Frequency;
314 }
315
316 /* Determine the integer part */
317 if ((USARTx->CR1 & USART_CR1_OVER8) != 0)
318 {
319 /* Integer part computing in case Oversampling mode is 8 Samples */
320 integerdivider = ((25 * apbclock) / (2 * (USART_InitStruct->USART_BaudRate)));
321 }
322 else /* if ((USARTx->CR1 & USART_CR1_OVER8) == 0) */
323 {
324 /* Integer part computing in case Oversampling mode is 16 Samples */
325 integerdivider = ((25 * apbclock) / (4 * (USART_InitStruct->USART_BaudRate)));
326 }
327 tmpreg = (integerdivider / 100) << 4;
328
329 /* Determine the fractional part */
330 fractionaldivider = integerdivider - (100 * (tmpreg >> 4));
331
332 /* Implement the fractional part in the register */
333 if ((USARTx->CR1 & USART_CR1_OVER8) != 0)
334 {
335 tmpreg |= ((((fractionaldivider * 8) + 50) / 100)) & ((uint8_t)0x07);
336 }
337 else /* if ((USARTx->CR1 & USART_CR1_OVER8) == 0) */
338 {
339 tmpreg |= ((((fractionaldivider * 16) + 50) / 100)) & ((uint8_t)0x0F);
340 }
341
342 /* Write to USART BRR register */
343 USARTx->BRR = (uint16_t)tmpreg;
344}

372void USART_ClockInit(USART_TypeDef* USARTx, USART_ClockInitTypeDef* USART_ClockInitStruct)
373{
374 uint32_t tmpreg = 0x00;
375 /* Check the parameters */
376 assert_param(IS_USART_1236_PERIPH(USARTx));
377 assert_param(IS_USART_CLOCK(USART_ClockInitStruct->USART_Clock));
378 assert_param(IS_USART_CPOL(USART_ClockInitStruct->USART_CPOL));
379 assert_param(IS_USART_CPHA(USART_ClockInitStruct->USART_CPHA));
380 assert_param(IS_USART_LASTBIT(USART_ClockInitStruct->USART_LastBit));
381
382/*---------------------------- USART CR2 Configuration -----------------------*/
383 tmpreg = USARTx->CR2;
384 /* Clear CLKEN, CPOL, CPHA and LBCL bits */
385 tmpreg &= (uint32_t)~((uint32_t)CR2_CLOCK_CLEAR_MASK);
386 /* Configure the USART Clock, CPOL, CPHA and LastBit ------------*/
387 /* Set CLKEN bit according to USART_Clock value */
388 /* Set CPOL bit according to USART_CPOL value */
389 /* Set CPHA bit according to USART_CPHA value */
390 /* Set LBCL bit according to USART_LastBit value */
391 tmpreg |= (uint32_t)USART_ClockInitStruct->USART_Clock | USART_ClockInitStruct->USART_CPOL |
392 USART_ClockInitStruct->USART_CPHA | USART_ClockInitStruct->USART_LastBit;
393 /* Write to USART CR2 */
394 USARTx->CR2 = (uint16_t)tmpreg;

1224void USART_ITConfig(USART_TypeDef* USARTx, uint16_t USART_IT, FunctionalState NewState)
1225{
1226 uint32_t usartreg = 0x00, itpos = 0x00, itmask = 0x00;
1227 uint32_t usartxbase = 0x00;
1228 /* Check the parameters */
1229 assert_param(IS_USART_ALL_PERIPH(USARTx));
1230 assert_param(IS_USART_CONFIG_IT(USART_IT));
1231 assert_param(IS_FUNCTIONAL_STATE(NewState));
1232
1233 /* The CTS interrupt is not available for UART4 and UART5 */
1234 if (USART_IT == USART_IT_CTS)
1235 {
1236 assert_param(IS_USART_1236_PERIPH(USARTx));
1237 }
1238
1239 usartxbase = (uint32_t)USARTx;
1240
1241 /* Get the USART register index */
1242 usartreg = (((uint8_t)USART_IT) >> 0x05);
1243
1244 /* Get the interrupt position */
1245 itpos = USART_IT & IT_MASK;
1246 itmask = (((uint32_t)0x01) << itpos);
1247
1248 if (usartreg == 0x01) /* The IT is in CR1 register */
1249 {
1250 usartxbase += 0x0C;
1251 }
1252 else if (usartreg == 0x02) /* The IT is in CR2 register */
1253 {
1254 usartxbase += 0x10;
1255 }
1256 else /* The IT is in CR3 register */
1257 {
1258 usartxbase += 0x14;
1259 }
1260 if (NewState != DISABLE)
1261 {
1262 *(__IO uint32_t*)usartxbase |= itmask;
1263 }
1264 else
1265 {
1266 *(__IO uint32_t*)usartxbase &= ~itmask;
1267 }
