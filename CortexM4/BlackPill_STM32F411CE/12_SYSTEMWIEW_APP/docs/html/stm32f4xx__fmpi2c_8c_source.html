160void FMPI2C_Init(FMPI2C_TypeDef* FMPI2Cx, FMPI2C_InitTypeDef* FMPI2C_InitStruct)
161{
162 uint32_t tmpreg = 0;
163
164 /* Check the parameters */
165 assert_param(IS_FMPI2C_ALL_PERIPH(FMPI2Cx));
166 assert_param(IS_FMPI2C_ANALOG_FILTER(FMPI2C_InitStruct->FMPI2C_AnalogFilter));
167 assert_param(IS_FMPI2C_DIGITAL_FILTER(FMPI2C_InitStruct->FMPI2C_DigitalFilter));
168 assert_param(IS_FMPI2C_MODE(FMPI2C_InitStruct->FMPI2C_Mode));
169 assert_param(IS_FMPI2C_OWN_ADDRESS1(FMPI2C_InitStruct->FMPI2C_OwnAddress1));
170 assert_param(IS_FMPI2C_ACK(FMPI2C_InitStruct->FMPI2C_Ack));
171 assert_param(IS_FMPI2C_ACKNOWLEDGE_ADDRESS(FMPI2C_InitStruct->FMPI2C_AcknowledgedAddress));
172
173 /* Disable FMPI2Cx Peripheral */
174 FMPI2Cx->CR1 &= (uint32_t)~((uint32_t)FMPI2C_CR1_PE);
175
176 /*---------------------------- FMPI2Cx FILTERS Configuration ------------------*/
177 /* Get the FMPI2Cx CR1 value */
178 tmpreg = FMPI2Cx->CR1;
179 /* Clear FMPI2Cx CR1 register */
180 tmpreg &= CR1_CLEAR_MASK;
181 /* Configure FMPI2Cx: analog and digital filter */
182 /* Set ANFOFF bit according to FMPI2C_AnalogFilter value */
183 /* Set DFN bits according to FMPI2C_DigitalFilter value */
184 tmpreg |= (uint32_t)FMPI2C_InitStruct->FMPI2C_AnalogFilter |(FMPI2C_InitStruct->FMPI2C_DigitalFilter << 8);
185
186 /* Write to FMPI2Cx CR1 */
187 FMPI2Cx->CR1 = tmpreg;
188
189 /*---------------------------- FMPI2Cx TIMING Configuration -------------------*/
190 /* Configure FMPI2Cx: Timing */
191 /* Set TIMINGR bits according to FMPI2C_Timing */
192 /* Write to FMPI2Cx TIMING */
193 FMPI2Cx->TIMINGR = FMPI2C_InitStruct->FMPI2C_Timing & TIMING_CLEAR_MASK;
194
195 /* Enable FMPI2Cx Peripheral */
196 FMPI2Cx->CR1 |= FMPI2C_CR1_PE;
197
198 /*---------------------------- FMPI2Cx OAR1 Configuration ---------------------*/
199 /* Clear tmpreg local variable */
200 tmpreg = 0;
201 /* Clear OAR1 register */
202 FMPI2Cx->OAR1 = (uint32_t)tmpreg;
203 /* Clear OAR2 register */
204 FMPI2Cx->OAR2 = (uint32_t)tmpreg;
205 /* Configure FMPI2Cx: Own Address1 and acknowledged address */
206 /* Set OA1MODE bit according to FMPI2C_AcknowledgedAddress value */
207 /* Set OA1 bits according to FMPI2C_OwnAddress1 value */
208 tmpreg = (uint32_t)((uint32_t)FMPI2C_InitStruct->FMPI2C_AcknowledgedAddress | \
209 (uint32_t)FMPI2C_InitStruct->FMPI2C_OwnAddress1);
210 /* Write to FMPI2Cx OAR1 */
211 FMPI2Cx->OAR1 = tmpreg;
212 /* Enable Own Address1 acknowledgement */
213 FMPI2Cx->OAR1 |= FMPI2C_OAR1_OA1EN;
214
215 /*---------------------------- FMPI2Cx MODE Configuration ---------------------*/
216 /* Configure FMPI2Cx: mode */
217 /* Set SMBDEN and SMBHEN bits according to FMPI2C_Mode value */
218 tmpreg = FMPI2C_InitStruct->FMPI2C_Mode;
219 /* Write to FMPI2Cx CR1 */
220 FMPI2Cx->CR1 |= tmpreg;
221
222 /*---------------------------- FMPI2Cx ACK Configuration ----------------------*/
223 /* Get the FMPI2Cx CR2 value */
224 tmpreg = FMPI2Cx->CR2;
225 /* Clear FMPI2Cx CR2 register */
226 tmpreg &= CR2_CLEAR_MASK;
227 /* Configure FMPI2Cx: acknowledgement */
228 /* Set NACK bit according to FMPI2C_Ack value */
229 tmpreg |= FMPI2C_InitStruct->FMPI2C_Ack;
230 /* Write to FMPI2Cx CR2 */
231 FMPI2Cx->CR2 = tmpreg;
232}

842{

859 (uint32_t)ReloadEndMode | (uint32_t)StartStopMode);
860
861 /* update CR2 register */
862 FMPI2Cx->CR2 = tmpreg;
