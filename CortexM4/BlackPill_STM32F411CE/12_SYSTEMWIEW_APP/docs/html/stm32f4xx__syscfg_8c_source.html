161void SYSCFG_EXTILineConfig(uint8_t EXTI_PortSourceGPIOx, uint8_t EXTI_PinSourcex)
162{
163 uint32_t tmp = 0x00;
164
165 /* Check the parameters */
166 assert_param(IS_EXTI_PORT_SOURCE(EXTI_PortSourceGPIOx));
167 assert_param(IS_EXTI_PIN_SOURCE(EXTI_PinSourcex));
168
169 tmp = ((uint32_t)0x0F) << (0x04 * (EXTI_PinSourcex & (uint8_t)0x03));
170 SYSCFG->EXTICR[EXTI_PinSourcex >> 0x02] &= ~tmp;
171 SYSCFG->EXTICR[EXTI_PinSourcex >> 0x02] |= (((uint32_t)EXTI_PortSourceGPIOx) << (0x04 * (EXTI_PinSourcex & (uint8_t)0x03)));
172}
