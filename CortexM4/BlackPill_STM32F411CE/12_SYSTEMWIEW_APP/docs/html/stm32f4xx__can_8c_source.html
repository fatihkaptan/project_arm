209uint8_t CAN_Init(CAN_TypeDef* CANx, CAN_InitTypeDef* CAN_InitStruct)
210{
211 uint8_t InitStatus = CAN_InitStatus_Failed;
212 uint32_t wait_ack = 0x00000000;
213 /* Check the parameters */
214 assert_param(IS_CAN_ALL_PERIPH(CANx));
215 assert_param(IS_FUNCTIONAL_STATE(CAN_InitStruct->CAN_TTCM));
216 assert_param(IS_FUNCTIONAL_STATE(CAN_InitStruct->CAN_ABOM));
217 assert_param(IS_FUNCTIONAL_STATE(CAN_InitStruct->CAN_AWUM));
218 assert_param(IS_FUNCTIONAL_STATE(CAN_InitStruct->CAN_NART));
219 assert_param(IS_FUNCTIONAL_STATE(CAN_InitStruct->CAN_RFLM));
220 assert_param(IS_FUNCTIONAL_STATE(CAN_InitStruct->CAN_TXFP));
221 assert_param(IS_CAN_MODE(CAN_InitStruct->CAN_Mode));
222 assert_param(IS_CAN_SJW(CAN_InitStruct->CAN_SJW));
223 assert_param(IS_CAN_BS1(CAN_InitStruct->CAN_BS1));
224 assert_param(IS_CAN_BS2(CAN_InitStruct->CAN_BS2));
225 assert_param(IS_CAN_PRESCALER(CAN_InitStruct->CAN_Prescaler));
226
227 /* Exit from sleep mode */
228 CANx->MCR &= (~(uint32_t)CAN_MCR_SLEEP);
229
230 /* Request initialisation */
231 CANx->MCR |= CAN_MCR_INRQ ;
232
233 /* Wait the acknowledge */
234 while (((CANx->MSR & CAN_MSR_INAK) != CAN_MSR_INAK) && (wait_ack != INAK_TIMEOUT))
235 {
236 wait_ack++;
237 }
238
239 /* Check acknowledge */
240 if ((CANx->MSR & CAN_MSR_INAK) != CAN_MSR_INAK)
241 {
242 InitStatus = CAN_InitStatus_Failed;
243 }
244 else
245 {
246 /* Set the time triggered communication mode */
247 if (CAN_InitStruct->CAN_TTCM == ENABLE)
248 {
249 CANx->MCR |= CAN_MCR_TTCM;
250 }
251 else
252 {
253 CANx->MCR &= ~(uint32_t)CAN_MCR_TTCM;
254 }
255
256 /* Set the automatic bus-off management */
257 if (CAN_InitStruct->CAN_ABOM == ENABLE)
258 {
259 CANx->MCR |= CAN_MCR_ABOM;
260 }
261 else
262 {
263 CANx->MCR &= ~(uint32_t)CAN_MCR_ABOM;
264 }
265
266 /* Set the automatic wake-up mode */
267 if (CAN_InitStruct->CAN_AWUM == ENABLE)
268 {
269 CANx->MCR |= CAN_MCR_AWUM;
270 }
271 else
272 {
273 CANx->MCR &= ~(uint32_t)CAN_MCR_AWUM;
274 }
275
276 /* Set the no automatic retransmission */
277 if (CAN_InitStruct->CAN_NART == ENABLE)
278 {
279 CANx->MCR |= CAN_MCR_NART;
280 }
281 else
282 {
283 CANx->MCR &= ~(uint32_t)CAN_MCR_NART;
284 }
285
286 /* Set the receive FIFO locked mode */
287 if (CAN_InitStruct->CAN_RFLM == ENABLE)
288 {
289 CANx->MCR |= CAN_MCR_RFLM;
290 }
291 else
292 {
293 CANx->MCR &= ~(uint32_t)CAN_MCR_RFLM;
294 }
295
296 /* Set the transmit FIFO priority */
297 if (CAN_InitStruct->CAN_TXFP == ENABLE)
298 {
299 CANx->MCR |= CAN_MCR_TXFP;
300 }
301 else
302 {
303 CANx->MCR &= ~(uint32_t)CAN_MCR_TXFP;
304 }
305
306 /* Set the bit timing register */
307 CANx->BTR = (uint32_t)((uint32_t)CAN_InitStruct->CAN_Mode << 30) | \
308 ((uint32_t)CAN_InitStruct->CAN_SJW << 24) | \
309 ((uint32_t)CAN_InitStruct->CAN_BS1 << 16) | \
310 ((uint32_t)CAN_InitStruct->CAN_BS2 << 20) | \
311 ((uint32_t)CAN_InitStruct->CAN_Prescaler - 1);
312
313 /* Request leave initialisation */
314 CANx->MCR &= ~(uint32_t)CAN_MCR_INRQ;
315
316 /* Wait the acknowledge */
317 wait_ack = 0;
318
319 while (((CANx->MSR & CAN_MSR_INAK) == CAN_MSR_INAK) && (wait_ack != INAK_TIMEOUT))
320 {
321 wait_ack++;
322 }
323
324 /* ...and check acknowledged */
325 if ((CANx->MSR & CAN_MSR_INAK) == CAN_MSR_INAK)
326 {
327 InitStatus = CAN_InitStatus_Failed;
328 }
329 else
330 {
331 InitStatus = CAN_InitStatus_Success ;
332 }
333 }
334
335 /* At this step, return the status of initialization */
336 return InitStatus;
337}

711uint8_t CAN_Transmit(CAN_TypeDef* CANx, CanTxMsg* TxMessage)
712{
713 uint8_t transmit_mailbox = 0;
714 /* Check the parameters */
715 assert_param(IS_CAN_ALL_PERIPH(CANx));
716 assert_param(IS_CAN_IDTYPE(TxMessage->IDE));
717 assert_param(IS_CAN_RTR(TxMessage->RTR));
718 assert_param(IS_CAN_DLC(TxMessage->DLC));
719
720 /* Select one empty transmit mailbox */
721 if ((CANx->TSR&CAN_TSR_TME0) == CAN_TSR_TME0)
722 {
723 transmit_mailbox = 0;
724 }
725 else if ((CANx->TSR&CAN_TSR_TME1) == CAN_TSR_TME1)
726 {
727 transmit_mailbox = 1;
728 }
729 else if ((CANx->TSR&CAN_TSR_TME2) == CAN_TSR_TME2)
730 {
731 transmit_mailbox = 2;
732 }
733 else
734 {
735 transmit_mailbox = CAN_TxStatus_NoMailBox;
736 }
737
738 if (transmit_mailbox != CAN_TxStatus_NoMailBox)
739 {
740 /* Set up the Id */
741 CANx->sTxMailBox[transmit_mailbox].TIR &= TMIDxR_TXRQ;
742 if (TxMessage->IDE == CAN_Id_Standard)
743 {
744 assert_param(IS_CAN_STDID(TxMessage->StdId));
745 CANx->sTxMailBox[transmit_mailbox].TIR |= ((TxMessage->StdId << 21) | \
746 TxMessage->RTR);
747 }
748 else
749 {
750 assert_param(IS_CAN_EXTID(TxMessage->ExtId));
751 CANx->sTxMailBox[transmit_mailbox].TIR |= ((TxMessage->ExtId << 3) | \
752 TxMessage->IDE | \
753 TxMessage->RTR);
754 }
755
756 /* Set up the DLC */
757 TxMessage->DLC &= (uint8_t)0x0000000F;
758 CANx->sTxMailBox[transmit_mailbox].TDTR &= (uint32_t)0xFFFFFFF0;
759 CANx->sTxMailBox[transmit_mailbox].TDTR |= TxMessage->DLC;
760
761 /* Set up the data field */
762 CANx->sTxMailBox[transmit_mailbox].TDLR = (((uint32_t)TxMessage->Data[3] << 24) |
763 ((uint32_t)TxMessage->Data[2] << 16) |
764 ((uint32_t)TxMessage->Data[1] << 8) |
765 ((uint32_t)TxMessage->Data[0]));
766 CANx->sTxMailBox[transmit_mailbox].TDHR = (((uint32_t)TxMessage->Data[7] << 24) |
767 ((uint32_t)TxMessage->Data[6] << 16) |
768 ((uint32_t)TxMessage->Data[5] << 8) |
769 ((uint32_t)TxMessage->Data[4]));
770 /* Request transmission */
771 CANx->sTxMailBox[transmit_mailbox].TIR |= TMIDxR_TXRQ;
772 }
773 return transmit_mailbox;
774}

784uint8_t CAN_TransmitStatus(CAN_TypeDef* CANx, uint8_t TransmitMailbox)
785{
786 uint32_t state = 0;
787
788 /* Check the parameters */
789 assert_param(IS_CAN_ALL_PERIPH(CANx));
790 assert_param(IS_CAN_TRANSMITMAILBOX(TransmitMailbox));
791
792 switch (TransmitMailbox)
793 {
794 case (CAN_TXMAILBOX_0):
795 state = CANx->TSR & (CAN_TSR_RQCP0 | CAN_TSR_TXOK0 | CAN_TSR_TME0);
796 break;
797 case (CAN_TXMAILBOX_1):
798 state = CANx->TSR & (CAN_TSR_RQCP1 | CAN_TSR_TXOK1 | CAN_TSR_TME1);
799 break;
800 case (CAN_TXMAILBOX_2):
801 state = CANx->TSR & (CAN_TSR_RQCP2 | CAN_TSR_TXOK2 | CAN_TSR_TME2);
802 break;
803 default:
804 state = CAN_TxStatus_Failed;
805 break;
806 }
807 switch (state)
808 {
809 /* transmit pending */
810 case (0x0): state = CAN_TxStatus_Pending;
811 break;
812 /* transmit failed */
813 case (CAN_TSR_RQCP0 | CAN_TSR_TME0): state = CAN_TxStatus_Failed;
814 break;
815 case (CAN_TSR_RQCP1 | CAN_TSR_TME1): state = CAN_TxStatus_Failed;
816 break;
817 case (CAN_TSR_RQCP2 | CAN_TSR_TME2): state = CAN_TxStatus_Failed;
818 break;
819 /* transmit succeeded */
820 case (CAN_TSR_RQCP0 | CAN_TSR_TXOK0 | CAN_TSR_TME0):state = CAN_TxStatus_Ok;
821 break;
822 case (CAN_TSR_RQCP1 | CAN_TSR_TXOK1 | CAN_TSR_TME1):state = CAN_TxStatus_Ok;
823 break;
824 case (CAN_TSR_RQCP2 | CAN_TSR_TXOK2 | CAN_TSR_TME2):state = CAN_TxStatus_Ok;
825 break;
826 default: state = CAN_TxStatus_Failed;
