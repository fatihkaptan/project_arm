297//
298// Determine how much we must pad the control block to make it a multiple of a cache line in size
299// Assuming: U8 = 1B
300// U16 = 2B
301// U32 = 4B
302// U8/U16/U32* = 4B
303//
304#if SEGGER_RTT_CPU_CACHE_LINE_SIZE // Avoid division by zero in case we do not have any cache
305 #define SEGGER_RTT__ROUND_UP_2_CACHE_LINE_SIZE(NumBytes) (((NumBytes + SEGGER_RTT_CPU_CACHE_LINE_SIZE - 1) / SEGGER_RTT_CPU_CACHE_LINE_SIZE) * SEGGER_RTT_CPU_CACHE_LINE_SIZE)
306#else
307 #define SEGGER_RTT__ROUND_UP_2_CACHE_LINE_SIZE(NumBytes) (NumBytes)
308#endif
309#define SEGGER_RTT__CB_SIZE (16 + 4 + 4 + (SEGGER_RTT_MAX_NUM_UP_BUFFERS * 24) + (SEGGER_RTT_MAX_NUM_DOWN_BUFFERS * 24))
310#define SEGGER_RTT__CB_PADDING (SEGGER_RTT__ROUND_UP_2_CACHE_LINE_SIZE(SEGGER_RTT__CB_SIZE) - SEGGER_RTT__CB_SIZE)

319//
320// Description for a circular buffer (also called "ring buffer")
321// which is used as up-buffer (T->H)
322//
323typedef struct {
324 const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
325 char* pBuffer; // Pointer to start of buffer
326 unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
327 unsigned WrOff; // Position of next item to be written by either target.
328 volatile unsigned RdOff; // Position of next item to be read by host. Must be volatile since it may be modified by host.
329 unsigned Flags; // Contains configuration flags. Flags[31:24] are used for validity check and must be zero. Flags[23:2] are reserved for future use. Flags[1:0] = RTT operating mode.
330} SEGGER_RTT_BUFFER_UP;
331
332//
333// Description for a circular buffer (also called "ring buffer")
334// which is used as down-buffer (H->T)
335//
336typedef struct {
337 const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
338 char* pBuffer; // Pointer to start of buffer
339 unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
340 volatile unsigned WrOff; // Position of next item to be written by host. Must be volatile since it may be modified by host.
341 unsigned RdOff; // Position of next item to be read by target (down-buffer).
342 unsigned Flags; // Contains configuration flags. Flags[31:24] are used for validity check and must be zero. Flags[23:2] are reserved for future use. Flags[1:0] = RTT operating mode.
343} SEGGER_RTT_BUFFER_DOWN;
344
345//
346// RTT control block which describes the number of buffers available
347// as well as the configuration for each buffer
348//
349//
350typedef struct {
351 char acID[16]; // Initialized to "SEGGER RTT"
352 int MaxNumUpBuffers; // Initialized to SEGGER_RTT_MAX_NUM_UP_BUFFERS (type. 2)
353 int MaxNumDownBuffers; // Initialized to SEGGER_RTT_MAX_NUM_DOWN_BUFFERS (type. 2)
354 SEGGER_RTT_BUFFER_UP aUp[SEGGER_RTT_MAX_NUM_UP_BUFFERS]; // Up buffers, transferring information up from target via debug probe to host
355 SEGGER_RTT_BUFFER_DOWN aDown[SEGGER_RTT_MAX_NUM_DOWN_BUFFERS]; // Down buffers, transferring information down from host via debug probe to target
356#if SEGGER_RTT__CB_PADDING
357 unsigned char aDummy[SEGGER_RTT__CB_PADDING];
358#endif
359} SEGGER_RTT_CB;
