808void PWR_EnterSTOPMode(uint32_t PWR_Regulator, uint8_t PWR_STOPEntry)
809{
810 uint32_t tmpreg = 0;
811
812 /* Check the parameters */
813 assert_param(IS_PWR_REGULATOR(PWR_Regulator));
814 assert_param(IS_PWR_STOP_ENTRY(PWR_STOPEntry));
815
816 /* Select the regulator state in STOP mode ---------------------------------*/
817 tmpreg = PWR->CR;
818 /* Clear PDDS and LPDS bits */
819 tmpreg &= CR_DS_MASK;
820
821 /* Set LPDS, MRLVDS and LPLVDS bits according to PWR_Regulator value */
822 tmpreg |= PWR_Regulator;
823
824 /* Store the new value */
825 PWR->CR = tmpreg;
826
827 /* Set SLEEPDEEP bit of Cortex System Control Register */
828 SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
829
830 /* Select STOP mode entry --------------------------------------------------*/
831 if(PWR_STOPEntry == PWR_STOPEntry_WFI)
832 {
833 /* Request Wait For Interrupt */
834 __WFI();
835 }
836 else
837 {
838 /* Request Wait For Event */
839 __WFE();
840 }
841 /* Reset SLEEPDEEP bit of Cortex System Control Register */
842 SCB->SCR &= (uint32_t)~((uint32_t)SCB_SCR_SLEEPDEEP_Msk);
843}
