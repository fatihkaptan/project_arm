281void TIM_TimeBaseInit(TIM_TypeDef* TIMx, TIM_TimeBaseInitTypeDef* TIM_TimeBaseInitStruct)
282{
283 uint16_t tmpcr1 = 0;
284
285 /* Check the parameters */
286 assert_param(IS_TIM_ALL_PERIPH(TIMx));
287 assert_param(IS_TIM_COUNTER_MODE(TIM_TimeBaseInitStruct->TIM_CounterMode));
288 assert_param(IS_TIM_CKD_DIV(TIM_TimeBaseInitStruct->TIM_ClockDivision));
289
290 tmpcr1 = TIMx->CR1;
291
292 if((TIMx == TIM1) || (TIMx == TIM8)||
293 (TIMx == TIM2) || (TIMx == TIM3)||
294 (TIMx == TIM4) || (TIMx == TIM5))
295 {
296 /* Select the Counter Mode */
297 tmpcr1 &= (uint16_t)(~(TIM_CR1_DIR | TIM_CR1_CMS));
298 tmpcr1 |= (uint32_t)TIM_TimeBaseInitStruct->TIM_CounterMode;
299 }
300
301 if((TIMx != TIM6) && (TIMx != TIM7))
302 {
303 /* Set the clock division */
304 tmpcr1 &= (uint16_t)(~TIM_CR1_CKD);
305 tmpcr1 |= (uint32_t)TIM_TimeBaseInitStruct->TIM_ClockDivision;
306 }
307
308 TIMx->CR1 = tmpcr1;
309
310 /* Set the Autoreload value */
311 TIMx->ARR = TIM_TimeBaseInitStruct->TIM_Period ;
312
313 /* Set the Prescaler value */
314 TIMx->PSC = TIM_TimeBaseInitStruct->TIM_Prescaler;
315
316 if ((TIMx == TIM1) || (TIMx == TIM8))
317 {
318 /* Set the Repetition Counter value */
319 TIMx->RCR = TIM_TimeBaseInitStruct->TIM_RepetitionCounter;
320 }
321
322 /* Generate an update event to reload the Prescaler
323 and the repetition counter(only for TIM1 and TIM8) value immediately */
324 TIMx->EGR = TIM_PSCReloadMode_Immediate;
325}
