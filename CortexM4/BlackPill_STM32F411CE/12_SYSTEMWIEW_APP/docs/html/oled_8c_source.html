367void OLED_SetPixel(int x, int y, COLOR_t c)
368{
369 unsigned char page, bitIdx, val;
370
371 x &= 0x7F;
372 y &= 0x3F;
373
374 page = y >> 3;
375 bitIdx = y & 7;
376
377 val = _DspRam[page * NSEG + x];
378
379 switch (c) {
380 case CLR_PIXEL:
381 val &= ~(1 << bitIdx);

383

403void OLED_Line(int x0, int y0, int x1, int y1, COLOR_t c)
404{
405 int steep, t ;
406 int deltax, deltay, error;
407 int x, y;
408 int ystep;
409
410 steep = abs(y1 - y0) > abs(x1 - x0);
411
412 if (steep)
413 { // swap x and y
414 t = x0; x0 = y0; y0 = t;
415 t = x1; x1 = y1; y1 = t;
416 }
417
418 if (x0 > x1)
419 { // swap ends
420 t = x0; x0 = x1; x1 = t;
421 t = y0; y0 = y1; y1 = t;
422 }
423
424 deltax = x1 - x0;
425 deltay = abs(y1 - y0);
426 error = 0;
427 y = y0;
428
429 if (y0 < y1)
430 ystep = 1;
431 else
432 ystep = -1;
433
434 for (x = x0; x < x1; x++)
435 {
436 if (steep)
437 OLED_SetPixel(y, x, c);
438 else
439 OLED_SetPixel(x, y, c);
440
441 error += deltay;
442 if ((error << 1) >= deltax)
443 {
444 y += ystep;
445 error -= deltax;
446 } // if
447 } // for
448} // line
449#endif

455{
456 float step, t;

459 step = PI / 2 / 64;
