468FLASH_Status FLASH_EraseSector(uint32_t FLASH_Sector, uint8_t VoltageRange)
469{
470 uint32_t tmp_psize = 0x0;
471 FLASH_Status status = FLASH_COMPLETE;
472
473 /* Check the parameters */
474 assert_param(IS_FLASH_SECTOR(FLASH_Sector));
475 assert_param(IS_VOLTAGERANGE(VoltageRange));
476
477 if(VoltageRange == VoltageRange_1)
478 {
479 tmp_psize = FLASH_PSIZE_BYTE;
480 }
481 else if(VoltageRange == VoltageRange_2)
482 {
483 tmp_psize = FLASH_PSIZE_HALF_WORD;
484 }
485 else if(VoltageRange == VoltageRange_3)
486 {
487 tmp_psize = FLASH_PSIZE_WORD;
488 }
489 else
490 {
491 tmp_psize = FLASH_PSIZE_DOUBLE_WORD;
492 }
493 /* Wait for last operation to be completed */
494 status = FLASH_WaitForLastOperation();
495
496 if(status == FLASH_COMPLETE)
497 {
498 /* if the previous operation is completed, proceed to erase the sector */
499 FLASH->CR &= CR_PSIZE_MASK;
500 FLASH->CR |= tmp_psize;
501 FLASH->CR &= SECTOR_MASK;
502 FLASH->CR |= FLASH_CR_SER | FLASH_Sector;
503 FLASH->CR |= FLASH_CR_STRT;
504
505 /* Wait for last operation to be completed */
506 status = FLASH_WaitForLastOperation();
507
508 /* if the erase operation is completed, disable the SER Bit */
509 FLASH->CR &= (~FLASH_CR_SER);
510 FLASH->CR &= SECTOR_MASK;
511 }
512 /* Return the Erase Status */
513 return status;
514}
