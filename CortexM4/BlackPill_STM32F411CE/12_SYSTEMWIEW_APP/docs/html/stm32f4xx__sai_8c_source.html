218 uint32_t tmpreg = 0;
219
220 /* Check the parameters */
221 assert_param(IS_SAI_BLOCK_PERIPH(SAI_Block_x));
222

224 assert_param(IS_SAI_BLOCK_MODE(SAI_InitStruct->SAI_AudioMode));
225 assert_param(IS_SAI_BLOCK_PROTOCOL(SAI_InitStruct->SAI_Protocol));
226 assert_param(IS_SAI_BLOCK_DATASIZE(SAI_InitStruct->SAI_DataSize));
227 assert_param(IS_SAI_BLOCK_FIRST_BIT(SAI_InitStruct->SAI_FirstBit));
228 assert_param(IS_SAI_BLOCK_CLOCK_STROBING(SAI_InitStruct->SAI_ClockStrobing));
229 assert_param(IS_SAI_BLOCK_SYNCHRO(SAI_InitStruct->SAI_Synchro));
230 assert_param(IS_SAI_BLOCK_SYNCEXT(SAI_InitStruct->SAI_SynchroExt));
231 assert_param(IS_SAI_BLOCK_OUTPUT_DRIVE(SAI_InitStruct->SAI_OUTDRIV));
232 assert_param(IS_SAI_BLOCK_NODIVIDER(SAI_InitStruct->SAI_NoDivider));
233 assert_param(IS_SAI_BLOCK_MASTER_DIVIDER(SAI_InitStruct->SAI_MasterDivider));
234 assert_param(IS_SAI_BLOCK_FIFO_THRESHOLD(SAI_InitStruct->SAI_FIFOThreshold));
235
236 /* SAI Block_x CR1 Configuration */
237 /* Get the SAI Block_x CR1 value */
238 tmpreg = SAI_Block_x->CR1;
239 /* Clear MODE, PRTCFG, DS, LSBFIRST, CKSTR, SYNCEN, OUTDRIV, NODIV, and MCKDIV bits */
240 tmpreg &= CR1_CLEAR_MASK;
241 /* Configure SAI_Block_x: Audio mode, Protocol, Data Size, first transmitted bit, Clock strobing
242 edge, Synchronization mode, Output drive, Master Divider and FIFO level */
243 /* Set MODE bits according to SAI_AudioMode value */
244 /* Set PRTCFG bits according to SAI_Protocol value */
245 /* Set DS bits according to SAI_DataSize value */
246 /* Set LSBFIRST bit according to SAI_FirstBit value */
247 /* Set CKSTR bit according to SAI_ClockStrobing value */
248 /* Set SYNCEN bit according to SAI_Synchro value */
249 /* Set OUTDRIV bit according to SAI_OUTDRIV value */
250 /* Set NODIV bit according to SAI_NoDivider value */
251 /* Set MCKDIV bits according to SAI_MasterDivider value */
252 tmpreg |= (uint32_t)(SAI_InitStruct->SAI_AudioMode | SAI_InitStruct->SAI_Protocol |
253 SAI_InitStruct->SAI_DataSize | SAI_InitStruct->SAI_FirstBit |
254 SAI_InitStruct->SAI_ClockStrobing | SAI_InitStruct->SAI_Synchro |
255 SAI_InitStruct->SAI_OUTDRIV | SAI_InitStruct->SAI_NoDivider |
256 SAI_InitStruct->SAI_SynchroExt | (uint32_t)((SAI_InitStruct->SAI_MasterDivider) << 20));
257 /* Write to SAI_Block_x CR1 */
258 SAI_Block_x->CR1 = tmpreg;
259
260 /* SAI Block_x CR2 Configuration */
261 /* Get the SAIBlock_x CR2 value */
262 tmpreg = SAI_Block_x->CR2;
263 /* Clear FTH bits */
264 tmpreg &= ~(SAI_xCR2_FTH);
265 /* Configure the FIFO Level */
266 /* Set FTH bits according to SAI_FIFOThreshold value */
267 tmpreg |= (uint32_t)(SAI_InitStruct->SAI_FIFOThreshold);
268 /* Write to SAI_Block_x CR2 */
269 SAI_Block_x->CR2 = tmpreg;
270}
