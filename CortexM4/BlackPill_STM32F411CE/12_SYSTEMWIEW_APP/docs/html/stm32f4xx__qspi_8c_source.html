212void QSPI_Init(QSPI_InitTypeDef* QSPI_InitStruct)
213{
214 uint32_t tmpreg = 0;
215
216 /* Check the QSPI parameters */
217 assert_param(IS_QSPI_SSHIFT(QSPI_InitStruct->QSPI_SShift));
218 assert_param(IS_QSPI_PRESCALER(QSPI_InitStruct->QSPI_Prescaler));
219 assert_param(IS_QSPI_CKMODE(QSPI_InitStruct->QSPI_CKMode));
220 assert_param(IS_QSPI_CSHTIME(QSPI_InitStruct->QSPI_CSHTime));
221 assert_param(IS_QSPI_FSIZE(QSPI_InitStruct->QSPI_FSize));
222 assert_param(IS_QSPI_FSEL(QSPI_InitStruct->QSPI_FSelect));
223 assert_param(IS_QSPI_DFM(QSPI_InitStruct->QSPI_DFlash));
224
225 /*------------------------ QSPI CR Configuration ------------------------*/
226 /* Get the QUADSPI CR1 value */
227 tmpreg = QUADSPI->CR;
228 /* Clear PRESCALER and SSHIFT bits */
229 tmpreg &= QSPI_CR_CLEAR_MASK;
230 /* Configure QUADSPI: Prescaler and Sample Shift */
231 tmpreg |= (uint32_t)(((QSPI_InitStruct->QSPI_Prescaler)<<24)
232 |(QSPI_InitStruct->QSPI_SShift)
233 |(QSPI_InitStruct->QSPI_FSelect)
234 |(QSPI_InitStruct->QSPI_DFlash));
235 /* Write to QUADSPI CR */
236 QUADSPI->CR = tmpreg;
237
238 /*------------------------ QUADSPI DCR Configuration ------------------------*/
239 /* Get the QUADSPI DCR value */
240 tmpreg = QUADSPI->DCR;
241 /* Clear FSIZE, CSHT and CKMODE bits */
242 tmpreg &= QSPI_DCR_CLEAR_MASK;
243 /* Configure QSPI: Flash Size, Chip Select High Time and Clock Mode */
244 tmpreg |= (uint32_t)(((QSPI_InitStruct->QSPI_FSize)<<16)
245 |(QSPI_InitStruct->QSPI_CSHTime)
246 |(QSPI_InitStruct->QSPI_CKMode));
247 /* Write to QSPI DCR */
248 QUADSPI->DCR = tmpreg;
249}
