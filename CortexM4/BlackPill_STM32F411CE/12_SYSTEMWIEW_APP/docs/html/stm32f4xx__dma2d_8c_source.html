121void DMA2D_Init(DMA2D_InitTypeDef* DMA2D_InitStruct)
122{
123
124 uint32_t outgreen = 0;
125 uint32_t outred = 0;
126 uint32_t outalpha = 0;
127 uint32_t pixline = 0;
128
129 /* Check the parameters */
130 assert_param(IS_DMA2D_MODE(DMA2D_InitStruct->DMA2D_Mode));
131 assert_param(IS_DMA2D_CMODE(DMA2D_InitStruct->DMA2D_CMode));
132 assert_param(IS_DMA2D_OGREEN(DMA2D_InitStruct->DMA2D_OutputGreen));
133 assert_param(IS_DMA2D_ORED(DMA2D_InitStruct->DMA2D_OutputRed));
134 assert_param(IS_DMA2D_OBLUE(DMA2D_InitStruct->DMA2D_OutputBlue));
135 assert_param(IS_DMA2D_OALPHA(DMA2D_InitStruct->DMA2D_OutputAlpha));
136 assert_param(IS_DMA2D_OUTPUT_OFFSET(DMA2D_InitStruct->DMA2D_OutputOffset));
137 assert_param(IS_DMA2D_LINE(DMA2D_InitStruct->DMA2D_NumberOfLine));
138 assert_param(IS_DMA2D_PIXEL(DMA2D_InitStruct->DMA2D_PixelPerLine));
139
140 /* Configures the DMA2D operation mode */
141 DMA2D->CR &= (uint32_t)CR_MASK;
142 DMA2D->CR |= (DMA2D_InitStruct->DMA2D_Mode);
143
144 /* Configures the color mode of the output image */
145 DMA2D->OPFCCR &= ~(uint32_t)DMA2D_OPFCCR_CM;
146 DMA2D->OPFCCR |= (DMA2D_InitStruct->DMA2D_CMode);
147
148 /* Configures the output color */
149
150 if (DMA2D_InitStruct->DMA2D_CMode == DMA2D_ARGB8888)
151 {
152 outgreen = DMA2D_InitStruct->DMA2D_OutputGreen << 8;
153 outred = DMA2D_InitStruct->DMA2D_OutputRed << 16;
154 outalpha = DMA2D_InitStruct->DMA2D_OutputAlpha << 24;
155 }
156 else
157
158 if (DMA2D_InitStruct->DMA2D_CMode == DMA2D_RGB888)
159 {
160 outgreen = DMA2D_InitStruct->DMA2D_OutputGreen << 8;
161 outred = DMA2D_InitStruct->DMA2D_OutputRed << 16;
162 outalpha = (uint32_t)0x00000000;
163 }
164
165 else
166
167 if (DMA2D_InitStruct->DMA2D_CMode == DMA2D_RGB565)
168 {
169 outgreen = DMA2D_InitStruct->DMA2D_OutputGreen << 5;
170 outred = DMA2D_InitStruct->DMA2D_OutputRed << 11;
171 outalpha = (uint32_t)0x00000000;
172 }
173
174 else
175
176 if (DMA2D_InitStruct->DMA2D_CMode == DMA2D_ARGB1555)
177 {
178 outgreen = DMA2D_InitStruct->DMA2D_OutputGreen << 5;
179 outred = DMA2D_InitStruct->DMA2D_OutputRed << 10;
180 outalpha = DMA2D_InitStruct->DMA2D_OutputAlpha << 15;
181 }
182
183 else /* DMA2D_CMode = DMA2D_ARGB4444 */
184 {
185 outgreen = DMA2D_InitStruct->DMA2D_OutputGreen << 4;
186 outred = DMA2D_InitStruct->DMA2D_OutputRed << 8;
187 outalpha = DMA2D_InitStruct->DMA2D_OutputAlpha << 12;
188 }
189 DMA2D->OCOLR = ((outgreen) | (outred) | (DMA2D_InitStruct->DMA2D_OutputBlue) | (outalpha));
190
191 /* Configures the output memory address */
192 DMA2D->OMAR = (DMA2D_InitStruct->DMA2D_OutputMemoryAdd);
193
194 /* Configure the line Offset */
195 DMA2D->OOR &= ~(uint32_t)DMA2D_OOR_LO;
196 DMA2D->OOR |= (DMA2D_InitStruct->DMA2D_OutputOffset);
197
198 /* Configure the number of line and pixel per line */
199 pixline = DMA2D_InitStruct->DMA2D_PixelPerLine << 16;
200 DMA2D->NLR &= ~(DMA2D_NLR_NL | DMA2D_NLR_PL);
201 DMA2D->NLR |= ((DMA2D_InitStruct->DMA2D_NumberOfLine) | (pixline));
