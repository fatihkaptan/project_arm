136void DCMI_Init(DCMI_InitTypeDef* DCMI_InitStruct)
137{
138 uint32_t temp = 0x0;
139
140 /* Check the parameters */
141 assert_param(IS_DCMI_CAPTURE_MODE(DCMI_InitStruct->DCMI_CaptureMode));
142 assert_param(IS_DCMI_SYNCHRO(DCMI_InitStruct->DCMI_SynchroMode));
143 assert_param(IS_DCMI_PCKPOLARITY(DCMI_InitStruct->DCMI_PCKPolarity));
144 assert_param(IS_DCMI_VSPOLARITY(DCMI_InitStruct->DCMI_VSPolarity));
145 assert_param(IS_DCMI_HSPOLARITY(DCMI_InitStruct->DCMI_HSPolarity));
146 assert_param(IS_DCMI_CAPTURE_RATE(DCMI_InitStruct->DCMI_CaptureRate));
147 assert_param(IS_DCMI_EXTENDED_DATA(DCMI_InitStruct->DCMI_ExtendedDataMode));
148
149 /* The DCMI configuration registers should be programmed correctly before
150 enabling the CR_ENABLE Bit and the CR_CAPTURE Bit */
151 DCMI->CR &= ~(DCMI_CR_ENABLE | DCMI_CR_CAPTURE);
152
153 /* Reset the old DCMI configuration */
154 temp = DCMI->CR;
155
156 temp &= ~((uint32_t)DCMI_CR_CM | DCMI_CR_ESS | DCMI_CR_PCKPOL |
157 DCMI_CR_HSPOL | DCMI_CR_VSPOL | DCMI_CR_FCRC_0 |
158 DCMI_CR_FCRC_1 | DCMI_CR_EDM_0 | DCMI_CR_EDM_1);
159
160 /* Sets the new configuration of the DCMI peripheral */
161 temp |= ((uint32_t)DCMI_InitStruct->DCMI_CaptureMode |
162 DCMI_InitStruct->DCMI_SynchroMode |
163 DCMI_InitStruct->DCMI_PCKPolarity |
164 DCMI_InitStruct->DCMI_VSPolarity |
165 DCMI_InitStruct->DCMI_HSPolarity |
166 DCMI_InitStruct->DCMI_CaptureRate |
167 DCMI_InitStruct->DCMI_ExtendedDataMode);
168
169 DCMI->CR = temp;
170}

407FlagStatus DCMI_GetFlagStatus(uint16_t DCMI_FLAG)
408{
409 FlagStatus bitstatus = RESET;
410 uint32_t dcmireg, tempreg = 0;
411
412 /* Check the parameters */
413 assert_param(IS_DCMI_GET_FLAG(DCMI_FLAG));
414
415 /* Get the DCMI register index */
416 dcmireg = (((uint16_t)DCMI_FLAG) >> 12);
417
418 if (dcmireg == 0x00) /* The FLAG is in RISR register */
419 {
420 tempreg= DCMI->RISR;
421 }
422 else if (dcmireg == 0x02) /* The FLAG is in SR register */
423 {
424 tempreg = DCMI->SR;
425 }
426 else /* The FLAG is in MISR register */
427 {
428 tempreg = DCMI->MISR;
429 }
430
431 if ((tempreg & DCMI_FLAG) != (uint16_t)RESET )
432 {
433 bitstatus = SET;
434 }
435 else
436 {
437 bitstatus = RESET;
438 }
439 /* Return the DCMI_FLAG status */
440 return bitstatus;
441}
