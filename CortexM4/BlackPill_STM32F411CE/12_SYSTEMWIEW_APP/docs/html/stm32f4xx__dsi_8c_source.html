106void DSI_DeInit(DSI_TypeDef *DSIx)
107{
108 /* Disable the DSI wrapper */
109 DSIx->WCR &= ~DSI_WCR_DSIEN;
110
111 /* Disable the DSI host */
112 DSIx->CR &= ~DSI_CR_EN;
113
114 /* D-PHY clock and digital disable */
115 DSIx->PCTLR &= ~(DSI_PCTLR_CKE | DSI_PCTLR_DEN);
116
117 /* Turn off the DSI PLL */
118 DSIx->WRPCR &= ~DSI_WRPCR_PLLEN;
119
120 /* Disable the regulator */
121 DSIx->WRPCR &= ~DSI_WRPCR_REGEN;
122
123 /* Check the parameters */
124 assert_param(IS_DSI_ALL_PERIPH(DSIx));
125 if(DSIx == DSI)
126 {
127 /* Enable DSI reset state */
128 RCC_APB2PeriphResetCmd(RCC_APB2Periph_DSI, ENABLE);
129 /* Release DSI from reset state */
130 RCC_APB2PeriphResetCmd(RCC_APB2Periph_DSI, DISABLE);

143void DSI_Init(DSI_TypeDef *DSIx,DSI_InitTypeDef* DSI_InitStruct, DSI_PLLInitTypeDef *PLLInit)
144{
145 uint32_t unitIntervalx4 = 0;
146 uint32_t tempIDF = 0;
147
148 /* Check function parameters */
149 assert_param(IS_DSI_PLL_NDIV(PLLInit->PLLNDIV));
150 assert_param(IS_DSI_PLL_IDF(PLLInit->PLLIDF));
151 assert_param(IS_DSI_PLL_ODF(PLLInit->PLLODF));
152 assert_param(IS_DSI_AUTO_CLKLANE_CONTROL(DSI_InitStruct->AutomaticClockLaneControl));
153 assert_param(IS_DSI_NUMBER_OF_LANES(DSI_InitStruct->NumberOfLanes));
154
155 /**************** Turn on the regulator and enable the DSI PLL ****************/
156
157 /* Enable the regulator */
158 DSIx->WRPCR |= DSI_WRPCR_REGEN;
159
160 /* Wait until the regulator is ready */
161 while(DSI_GetFlagStatus(DSIx, DSI_FLAG_RRS) == RESET )
162 {}
163
164 /* Set the PLL division factors */
165 DSIx->WRPCR &= ~(DSI_WRPCR_PLL_NDIV | DSI_WRPCR_PLL_IDF | DSI_WRPCR_PLL_ODF);
166 DSIx->WRPCR |= (((PLLInit->PLLNDIV)<<2) | ((PLLInit->PLLIDF)<<11) | ((PLLInit->PLLODF)<<16));
167
168 /* Enable the DSI PLL */
169 DSIx->WRPCR |= DSI_WRPCR_PLLEN;
170
171 /* Wait for the lock of the PLL */
172 while(DSI_GetFlagStatus(DSIx, DSI_FLAG_PLLLS) == RESET)
173 {}
174
175 /*************************** Set the PHY parameters ***************************/
176
177 /* D-PHY clock and digital enable*/
178 DSIx->PCTLR |= (DSI_PCTLR_CKE | DSI_PCTLR_DEN);
179
180 /* Clock lane configuration */
181 DSIx->CLCR &= ~(DSI_CLCR_DPCC | DSI_CLCR_ACR);
182 DSIx->CLCR |= (DSI_CLCR_DPCC | DSI_InitStruct->AutomaticClockLaneControl);
183
184 /* Configure the number of active data lanes */
185 DSIx->PCONFR &= ~DSI_PCONFR_NL;
186 DSIx->PCONFR |= DSI_InitStruct->NumberOfLanes;
187
188 /************************ Set the DSI clock parameters ************************/
189 /* Set the TX escape clock division factor */
190 DSIx->CCR &= ~DSI_CCR_TXECKDIV;
191 DSIx->CCR = DSI_InitStruct->TXEscapeCkdiv;
192
193 /* Calculate the bit period in high-speed mode in unit of 0.25 ns (UIX4) */
194 /* The equation is : UIX4 = IntegerPart( (1000/F_PHY_Mhz) * 4 ) */
195 /* Where : F_PHY_Mhz = (NDIV * HSE_Mhz) / (IDF * ODF) */
196 tempIDF = (PLLInit->PLLIDF > 0) ? PLLInit->PLLIDF : 1;
197 unitIntervalx4 = (4000000 * tempIDF * (1 << PLLInit->PLLODF)) / ((HSE_VALUE/1000) * PLLInit->PLLNDIV);
198
199 /* Set the bit period in high-speed mode */
200 DSIx->WPCR[0] &= ~DSI_WPCR0_UIX4;
201 DSIx->WPCR[0] |= unitIntervalx4;
202
203 /****************************** Error management *****************************/
204 /* Disable all error interrupts */
205 DSIx->IER[0] = 0;
206 DSIx->IER[1] = 0;
207}

266{
267 /* Check the parameters */
268 assert_param(IS_DSI_COLOR_CODING(VidCfg->ColorCoding));
269 assert_param(IS_DSI_VIDEO_MODE_TYPE(VidCfg->Mode));
270 assert_param(IS_DSI_LP_COMMAND(VidCfg->LPCommandEnable));
271 assert_param(IS_DSI_LP_HFP(VidCfg->LPHorizontalFrontPorchEnable));
272 assert_param(IS_DSI_LP_HBP(VidCfg->LPHorizontalBackPorchEnable));
273 assert_param(IS_DSI_LP_VACTIVE(VidCfg->LPVerticalActiveEnable));
274 assert_param(IS_DSI_LP_VFP(VidCfg->LPVerticalFrontPorchEnable));
275 assert_param(IS_DSI_LP_VBP(VidCfg->LPVerticalBackPorchEnable));
276 assert_param(IS_DSI_LP_VSYNC(VidCfg->LPVerticalSyncActiveEnable));
277 assert_param(IS_DSI_FBTAA(VidCfg->FrameBTAAcknowledgeEnable));
278 assert_param(IS_DSI_DE_POLARITY(VidCfg->DEPolarity));
279 assert_param(IS_DSI_VSYNC_POLARITY(VidCfg->VSPolarity));
280 assert_param(IS_DSI_HSYNC_POLARITY(VidCfg->HSPolarity));
281 /* Check the LooselyPacked variant only in 18-bit mode */
282 if(VidCfg->ColorCoding == DSI_RGB666)
283 {
284 assert_param(IS_DSI_LOOSELY_PACKED(VidCfg->LooselyPacked));
285 }
286
287 /* Select video mode by resetting CMDM and DSIM bits */
288 DSIx->MCR &= ~DSI_MCR_CMDM;
289 DSIx->WCFGR &= ~DSI_WCFGR_DSIM;
290
291 /* Configure the video mode transmission type */
292 DSIx->VMCR &= ~DSI_VMCR_VMT;
293 DSIx->VMCR |= VidCfg->Mode;
294
295 /* Configure the video packet size */
296 DSIx->VPCR &= ~DSI_VPCR_VPSIZE;
297 DSIx->VPCR |= VidCfg->PacketSize;
298
299 /* Set the chunks number to be transmitted through the DSI link */
300 DSIx->VCCR &= ~DSI_VCCR_NUMC;
301 DSIx->VCCR |= VidCfg->NumberOfChunks;
302
303 /* Set the size of the null packet */
304 DSIx->VNPCR &= ~DSI_VNPCR_NPSIZE;
305 DSIx->VNPCR |= VidCfg->NullPacketSize;
306
307 /* Select the virtual channel for the LTDC interface traffic */
308 DSIx->LVCIDR &= ~DSI_LVCIDR_VCID;
309 DSIx->LVCIDR |= VidCfg->VirtualChannelID;
310
311 /* Configure the polarity of control signals */
312 DSIx->LPCR &= ~(DSI_LPCR_DEP | DSI_LPCR_VSP | DSI_LPCR_HSP);
313 DSIx->LPCR |= (VidCfg->DEPolarity | VidCfg->VSPolarity | VidCfg->HSPolarity);
314
315 /* Select the color coding for the host */
316 DSIx->LCOLCR &= ~DSI_LCOLCR_COLC;
317 DSIx->LCOLCR |= VidCfg->ColorCoding;
318
319 /* Select the color coding for the wrapper */
320 DSIx->WCFGR &= ~DSI_WCFGR_COLMUX;
321 DSIx->WCFGR |= ((VidCfg->ColorCoding)<<1);
322
323 /* Enable/disable the loosely packed variant to 18-bit configuration */
324 if(VidCfg->ColorCoding == DSI_RGB666)
325 {
326 DSIx->LCOLCR &= ~DSI_LCOLCR_LPE;
327 DSIx->LCOLCR |= VidCfg->LooselyPacked;
328 }
329
330 /* Set the Horizontal Synchronization Active (HSA) in lane byte clock cycles */
331 DSIx->VHSACR &= ~DSI_VHSACR_HSA;
332 DSIx->VHSACR |= VidCfg->HorizontalSyncActive;
333
334 /* Set the Horizontal Back Porch (HBP) in lane byte clock cycles */
335 DSIx->VHBPCR &= ~DSI_VHBPCR_HBP;
336 DSIx->VHBPCR |= VidCfg->HorizontalBackPorch;
337
338 /* Set the total line time (HLINE=HSA+HBP+HACT+HFP) in lane byte clock cycles */

340 DSIx->VLCR |= VidCfg->HorizontalLine;

349
350 /* Set the Vertical Front Porch (VFP)*/
351 DSIx->VVFPCR &= ~DSI_VVFPCR_VFP;
352 DSIx->VVFPCR |= VidCfg->VerticalFrontPorch;
353
354 /* Set the Vertical Active period*/
355 DSIx->VVACR &= ~DSI_VVACR_VA;
356 DSIx->VVACR |= VidCfg->VerticalActive;
357
358 /* Configure the command transmission mode */
359 DSIx->VMCR &= ~DSI_VMCR_LPCE;
360 DSIx->VMCR |= VidCfg->LPCommandEnable;
361
362 /* Low power largest packet size */
363 DSIx->LPMCR &= ~DSI_LPMCR_LPSIZE;
364 DSIx->LPMCR |= ((VidCfg->LPLargestPacketSize)<<16);
365
366 /* Low power VACT largest packet size */
367 DSIx->LPMCR &= ~DSI_LPMCR_VLPSIZE;
368 DSIx->LPMCR |= VidCfg->LPVACTLargestPacketSize;
369
370 /* Enable LP transition in HFP period */
371 DSIx->VMCR &= ~DSI_VMCR_LPHFPE;
372 DSIx->VMCR |= VidCfg->LPHorizontalFrontPorchEnable;
373
374 /* Enable LP transition in HBP period */

376 DSIx->VMCR |= VidCfg->LPHorizontalBackPorchEnable;
377
378 /* Enable LP transition in VACT period */

382 /* Enable LP transition in VFP period */

537void DSI_ConfigPhyTimer(DSI_TypeDef *DSIx, DSI_PHY_TimerTypeDef *PhyTimers)
538{
539 uint32_t maxTime = 0;
540
541 maxTime = (PhyTimers->ClockLaneLP2HSTime > PhyTimers->ClockLaneHS2LPTime)? PhyTimers->ClockLaneLP2HSTime: PhyTimers->ClockLaneHS2LPTime;
542
543 /* Clock lane timer configuration */
544 /* In Automatic Clock Lane control mode, the DSI Host can turn off the clock lane between two
545 High-Speed transmission.
546 To do so, the DSI Host calculates the time required for the clock lane to change from HighSpeed
547 to Low-Power and from Low-Power to High-Speed.
548 This timings are configured by the HS2LP_TIME and LP2HS_TIME in the DSI Host Clock Lane Timer Configuration Register (DSI_CLTCR).
549 But the DSI Host is not calculating LP2HS_TIME + HS2LP_TIME but 2 x HS2LP_TIME.
550
551 Workaround : Configure HS2LP_TIME and LP2HS_TIME with the same value being the max of HS2LP_TIME or LP2HS_TIME.
552 */
553 DSIx->CLTCR &= ~(DSI_CLTCR_LP2HS_TIME | DSI_CLTCR_HS2LP_TIME);
554 DSIx->CLTCR |= (maxTime | ((maxTime)<<16));
555
556 /* Data lane timer configuration */
557 DSIx->DLTCR &= ~(DSI_DLTCR_MRD_TIME | DSI_DLTCR_LP2HS_TIME | DSI_DLTCR_HS2LP_TIME);
558 DSIx->DLTCR |= (PhyTimers->DataLaneMaxReadTime | ((PhyTimers->DataLaneLP2HSTime)<<16) | ((PhyTimers->DataLaneHS2LPTime)<<24));
559
560 /* Configure the wait period to request HS transmission after a stop state */
561 DSIx->PCONFR &= ~DSI_PCONFR_SW_TIME;
562 DSIx->PCONFR |= ((PhyTimers->StopWaitTime)<<8);
563}

746void DSI_LongWrite(DSI_TypeDef *DSIx,
747 uint32_t ChannelID,
748 uint32_t Mode,
749 uint32_t NbParams,
750 uint32_t Param1,
751 uint8_t* ParametersTable)
752{
753 uint32_t uicounter = 0;
754
755 /* Check the parameters */
756 assert_param(IS_DSI_LONG_WRITE_PACKET_TYPE(Mode));
757
758 /* Wait for Command FIFO Empty */
759 while((DSIx->GPSR & DSI_GPSR_CMDFE) == 0)
760 {}
761
762 /* Set the DCS code hexadecimal on payload byte 1, and the other parameters on the write FIFO command*/
763 while(uicounter < NbParams)
764 {
765 if(uicounter == 0x00)
766 {
767 DSIx->GPDR=(Param1 | \
768 ((uint32_t)(*(ParametersTable+uicounter))<<8) | \
769 ((uint32_t)(*(ParametersTable+uicounter+1))<<16) | \
770 ((uint32_t)(*(ParametersTable+uicounter+2))<<24));
771 uicounter += 3;
772 }
773 else
774 {
775 DSIx->GPDR=((*(ParametersTable+uicounter)) | \
776 ((uint32_t)(*(ParametersTable+uicounter+1))<<8) | \
777 ((uint32_t)(*(ParametersTable+uicounter+2))<<16) | \
778 ((uint32_t)(*(ParametersTable+uicounter+3))<<24));
779 uicounter+=4;
780 }
781 }
782
783 /* Configure the packet to send a long DCS command */
784 DSI_ConfigPacketHeader(DSIx,
785 ChannelID,
786 Mode,
787 ((NbParams+1)&0x00FF),
788 (((NbParams+1)&0xFF00)>>8));
789}

803void DSI_Read(DSI_TypeDef *DSIx,
804 uint32_t ChannelNbr,
805 uint8_t* Array,
806 uint32_t Size,
807 uint32_t Mode,
808 uint32_t DCSCmd,
809 uint8_t* ParametersTable)
810{
811
812 /* Check the parameters */
813 assert_param(IS_DSI_READ_PACKET_TYPE(Mode));
814
815 if(Size > 2)
816 {
817 /* set max return packet size */
818 DSI_ShortWrite(DSIx, ChannelNbr, DSI_MAX_RETURN_PKT_SIZE, ((Size)&0xFF), (((Size)>>8)&0xFF));
819 }
820
821 /* Configure the packet to read command */
822 if (Mode == DSI_DCS_SHORT_PKT_READ)
823 {
824 DSI_ConfigPacketHeader(DSIx, ChannelNbr, Mode, DCSCmd, 0);
825 }
826 else if (Mode == DSI_GEN_SHORT_PKT_READ_P0)
827 {
828 DSI_ConfigPacketHeader(DSIx, ChannelNbr, Mode, 0, 0);
829 }
830 else if (Mode == DSI_GEN_SHORT_PKT_READ_P1)
831 {
832 DSI_ConfigPacketHeader(DSIx, ChannelNbr, Mode, ParametersTable[0], 0);
833 }
834 else /* DSI_GEN_SHORT_PKT_READ_P2 */
835 {
836 DSI_ConfigPacketHeader(DSIx, ChannelNbr, Mode, ParametersTable[0], ParametersTable[1]);
837 }
838
839 /* Check that the payload read FIFO is not empty */
840 while((DSIx->GPSR & DSI_GPSR_PRDFE) == DSI_GPSR_PRDFE)
841 {}
842
843 /* Get the first byte */
844 *((uint32_t *)Array) = (DSIx->GPDR);
845 if (Size > 4)
846 {
847 Size -= 4;
848 Array += 4;
849 }
850
851 /* Get the remaining bytes if any */
852 while(((int)(Size)) > 0)
853 {
854 if((DSIx->GPSR & DSI_GPSR_PRDFE) == 0)
855 {
856 *((uint32_t *)Array) = (DSIx->GPDR);
857 Size -= 4;
858 Array += 4;
859 }
860 }
861}

961void DSI_EnterULPM(DSI_TypeDef *DSIx)
962{
963 /* Clock lane configuration: no more HS request */
964 DSIx->CLCR &= ~DSI_CLCR_DPCC;
965
966 /* Use system PLL as byte lane clock source before stopping DSIPHY clock source */
967 RCC_DSIClockSourceConfig(RCC_DSICLKSource_PLLR);
968
969 /* ULPS Request on Clock and Data Lanes */
970 DSIx->PUCR |= (DSI_PUCR_URCL | DSI_PUCR_URDL);
971
972 /* Wait until all active lanes exit ULPM */
973 if((DSIx->PCONFR & DSI_PCONFR_NL) == DSI_ONE_DATA_LANE)
974 {
975 while((DSIx->PSR & (DSI_PSR_UAN0 | DSI_PSR_UANC)) != 0)
976 {}
977 }
978 else /* DSI_TWO_DATA_LANES */
979 {
980 while((DSIx->PSR & (DSI_PSR_UAN0 | DSI_PSR_UAN1 | DSI_PSR_UANC)) != 0)
981 {}
982 }
983
984 /* Turn off the DSI PLL */
985 DSIx->WRPCR &= ~DSI_WRPCR_PLLEN;
986}

994void DSI_ExitULPM(DSI_TypeDef *DSIx)
995{
996 /* Turn on the DSI PLL */
997 DSIx->WRPCR |= DSI_WRPCR_PLLEN;
998
999 /* Wait for the lock of the PLL */
1000 while(DSI_GetFlagStatus(DSIx, DSI_FLAG_PLLLS) == RESET)
1001 {}
1002
1003 /* Exit ULPS on Clock and Data Lanes */
1004 DSIx->PUCR |= (DSI_PUCR_UECL | DSI_PUCR_UEDL);
1005
1006 /* Wait until all active lanes exit ULPM */
1007 if((DSIx->PCONFR & DSI_PCONFR_NL) == DSI_ONE_DATA_LANE)
1008 {
1009 while((DSIx->PSR & (DSI_PSR_UAN0 | DSI_PSR_UANC)) != (DSI_PSR_UAN0 | DSI_PSR_UANC))
1010 {}
1011 }
1012 else /* DSI_TWO_DATA_LANES */
1013 {
1014 while((DSIx->PSR & (DSI_PSR_UAN0 | DSI_PSR_UAN1 | DSI_PSR_UANC)) != (DSI_PSR_UAN0 | DSI_PSR_UAN1 | DSI_PSR_UANC))
1015 {}
1016 }
1017
1018 /* De-assert the ULPM requests and the ULPM exit bits */
1019 DSIx->PUCR = 0;
1020
1021 /* Switch the lanbyteclock source in the RCC from system PLL to D-PHY */
1022 RCC_DSIClockSourceConfig(RCC_DSICLKSource_PHY);
1023
1024 /* Restore clock lane configuration to HS */
1025 DSIx->CLCR |= DSI_CLCR_DPCC;
1026}
