139void LTDC_Init(LTDC_InitTypeDef* LTDC_InitStruct)

144 uint32_t totalwidth = 0;
145 uint32_t backgreen = 0;

148 /* Check function parameters */
149 assert_param(IS_LTDC_HSYNC(LTDC_InitStruct->LTDC_HorizontalSync));
150 assert_param(IS_LTDC_VSYNC(LTDC_InitStruct->LTDC_VerticalSync));

160 assert_param(IS_LTDC_PCPOL(LTDC_InitStruct->LTDC_PCPolarity));
161 assert_param(IS_LTDC_BackBlueValue(LTDC_InitStruct->LTDC_BackgroundBlueValue));
162 assert_param(IS_LTDC_BackGreenValue(LTDC_InitStruct->LTDC_BackgroundGreenValue));
163 assert_param(IS_LTDC_BackRedValue(LTDC_InitStruct->LTDC_BackgroundRedValue));
164
165 /* Sets Synchronization size */
166 LTDC->SSCR &= ~(LTDC_SSCR_VSH | LTDC_SSCR_HSW);
167 horizontalsync = (LTDC_InitStruct->LTDC_HorizontalSync << 16);
168 LTDC->SSCR |= (horizontalsync | LTDC_InitStruct->LTDC_VerticalSync);
169
170 /* Sets Accumulated Back porch */
171 LTDC->BPCR &= ~(LTDC_BPCR_AVBP | LTDC_BPCR_AHBP);
172 accumulatedHBP = (LTDC_InitStruct->LTDC_AccumulatedHBP << 16);
173 LTDC->BPCR |= (accumulatedHBP | LTDC_InitStruct->LTDC_AccumulatedVBP);
174
175 /* Sets Accumulated Active Width */
176 LTDC->AWCR &= ~(LTDC_AWCR_AAH | LTDC_AWCR_AAW);
177 accumulatedactiveW = (LTDC_InitStruct->LTDC_AccumulatedActiveW << 16);
178 LTDC->AWCR |= (accumulatedactiveW | LTDC_InitStruct->LTDC_AccumulatedActiveH);
179
180 /* Sets Total Width */
181 LTDC->TWCR &= ~(LTDC_TWCR_TOTALH | LTDC_TWCR_TOTALW);
182 totalwidth = (LTDC_InitStruct->LTDC_TotalWidth << 16);
183 LTDC->TWCR |= (totalwidth | LTDC_InitStruct->LTDC_TotalHeigh);
184
185 LTDC->GCR &= (uint32_t)GCR_MASK;
186 LTDC->GCR |= (uint32_t)(LTDC_InitStruct->LTDC_HSPolarity | LTDC_InitStruct->LTDC_VSPolarity | \
187 LTDC_InitStruct->LTDC_DEPolarity | LTDC_InitStruct->LTDC_PCPolarity);
188
189 /* sets the background color value */
190 backgreen = (LTDC_InitStruct->LTDC_BackgroundGreenValue << 8);
191 backred = (LTDC_InitStruct->LTDC_BackgroundRedValue << 16);
192
193 LTDC->BCCR &= ~(LTDC_BCCR_BCBLUE | LTDC_BCCR_BCGREEN | LTDC_BCCR_BCRED);
194 LTDC->BCCR |= (backred | backgreen | LTDC_InitStruct->LTDC_BackgroundBlueValue);

700void LTDC_LayerPosition(LTDC_Layer_TypeDef* LTDC_Layerx, uint16_t OffsetX, uint16_t OffsetY)
701{
702
703 uint32_t tempreg, temp;
704 uint32_t horizontal_start;
705 uint32_t horizontal_stop;
706 uint32_t vertical_start;
707 uint32_t vertical_stop;
708
709 LTDC_Layerx->WHPCR &= ~(LTDC_LxWHPCR_WHSTPOS | LTDC_LxWHPCR_WHSPPOS);
710 LTDC_Layerx->WVPCR &= ~(LTDC_LxWVPCR_WVSTPOS | LTDC_LxWVPCR_WVSPPOS);
711
712 /* Reconfigures the horizontal and vertical start position */
713 tempreg = LTDC->BPCR;
714 horizontal_start = (tempreg >> 16) + 1 + OffsetX;
715 vertical_start = (tempreg & 0xFFFF) + 1 + OffsetY;
716
717 /* Reconfigures the horizontal and vertical stop position */
718 /* Get the number of byte per pixel */
719
720 tempreg = LTDC_Layerx->PFCR;
721
722 if (tempreg == LTDC_Pixelformat_ARGB8888)
723 {
724 temp = 4;
725 }
726 else if (tempreg == LTDC_Pixelformat_RGB888)
727 {
728 temp = 3;
729 }
730 else if ((tempreg == LTDC_Pixelformat_ARGB4444) ||
731 (tempreg == LTDC_Pixelformat_RGB565) ||
732 (tempreg == LTDC_Pixelformat_ARGB1555) ||
733 (tempreg == LTDC_Pixelformat_AL88))
734 {
735 temp = 2;
736 }
737 else
738 {
739 temp = 1;
740 }
741
742 tempreg = LTDC_Layerx->CFBLR;
743 horizontal_stop = (((tempreg & 0x1FFF) - 3)/temp) + horizontal_start - 1;
744
745 tempreg = LTDC_Layerx->CFBLNR;
746 vertical_stop = (tempreg & 0x7FF) + vertical_start - 1;
747
748 LTDC_Layerx->WHPCR = horizontal_start | (horizontal_stop << 16);
749 LTDC_Layerx->WVPCR = vertical_start | (vertical_stop << 16);
750}
